1   package nl.tudelft.dnainator.javafx.controllers;
2   
3   import nl.tudelft.dnainator.core.PropertyType;
4   import nl.tudelft.dnainator.javafx.widgets.Propertyable;
5   import nl.tudelft.dnainator.javafx.widgets.animations.LeftSlideAnimation;
6   import nl.tudelft.dnainator.javafx.widgets.animations.SlidingAnimation;
7   import nl.tudelft.dnainator.javafx.widgets.animations.TransitionAnimation.Position;
8   import javafx.fxml.FXML;
9   import javafx.scene.Node;
10  import javafx.scene.control.Label;
11  import javafx.scene.control.ScrollPane;
12  import javafx.scene.layout.VBox;
13  
14  /**
15   * Controls the property pane on the right side of the application.
16   * It provides information about selected strain data.
17   */
18  public class PropertyPaneController {
19  	private SlidingAnimation animation;
20  	private static final int WIDTH = 300;
21  	private static final int ANIM_DURATION = 250;
22  	private static final int MAX_PROPERTY_SIZE = 750;
23  
24  	@SuppressWarnings("unused") @FXML private VBox propertyPane;
25  
26  	/*
27  	 * Sets up the animation for the property pane.
28  	 */
29  	@SuppressWarnings("unused") @FXML
30  	private void initialize() {
31  		animation = new LeftSlideAnimation(propertyPane, WIDTH, ANIM_DURATION, Position.RIGHT);
32  	}
33  
34  	/**
35  	 * Updates the displayed information.
36  	 * @param p The new {@link Propertyable} whose information to display.
37  	 */
38  	public void update(Propertyable p) {
39  		onClickOpenPane();
40  		propertyPane.getChildren().clear();
41  		p.getPropertyMap().forEach((k, v) -> addLabel(k, v));
42  	}
43  
44  	private void addLabel(PropertyType type, String value) {
45  		Label description = new Label(type.description());
46  		description.getStyleClass().add("property-header");
47  
48  		Node label = new Label(value);
49  		label.getStyleClass().add("property-info");
50  		if (value != null && value.length() > MAX_PROPERTY_SIZE) {
51  			ScrollPane sp = new ScrollPane(label);
52  			sp.setFitToWidth(true);
53  			sp.setPrefHeight(WIDTH);
54  			label = sp;
55  		}
56  		propertyPane.getChildren().add(new VBox(description, label));
57  	}
58  
59  	/**
60  	 * Toggle the sliding animation on the property pane.
61  	 */
62  	public void toggle() {
63  		animation.toggle();
64  	}
65  	
66  	/**
67  	 * Opens the {@link PropertyPane} when a drawable is clicked and it was not open.
68  	 */
69  	private void onClickOpenPane() {
70  		if (!propertyPane.isVisible()) {
71  			toggle();
72  		}
73  	}
74  }
