1   package nl.tudelft.dnainator.javafx.drawables.annotations;
2   
3   import javafx.scene.Node;
4   import javafx.scene.control.Tooltip;
5   import javafx.scene.layout.VBox;
6   import javafx.scene.text.Text;
7   import nl.tudelft.dnainator.annotation.Annotation;
8   import nl.tudelft.dnainator.javafx.drawables.Drawable;
9   
10  /**
11   * An implementation of {@link Drawable} to display a gene from an {@link Annotation}.
12   */
13  public class Gene extends VBox implements Drawable {
14  
15  	/**
16  	 * Instantiates a new {@link Gene}.
17  	 * @param annotation The {@link Annotation} to display.
18  	 */
19  	public Gene(Annotation annotation) {
20  		draw(annotation);
21  	}
22  
23  	private void draw(Annotation annotation) {
24  		Text name = new Text(annotation.getGeneName());
25  		Tooltip tooltip = new Tooltip("Coordinates: " + annotation.getStart() + " to "
26  				+ annotation.getEnd() + "\n" + "Sense: " + annotation.isSense());
27  		Tooltip.install(name, tooltip);
28  		getChildren().add(name);
29  		if (annotation.isMutation()) {
30  			getStyleClass().add("dr-mutation");
31  		} else {
32  			getStyleClass().add("annotation");
33  		}
34  	}
35  
36  	@Override
37  	public void addStyle(String style) {
38  		for (Node child : getChildren()) {
39  			child.getStyleClass().add(style);
40  		}
41  	}
42  
43  	@Override
44  	public void removeStyle(String style) {
45  		for (Node child : getChildren()) {
46  			child.getStyleClass().remove(style);
47  		}
48  	}
49  }
