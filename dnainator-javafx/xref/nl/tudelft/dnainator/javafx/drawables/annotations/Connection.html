1   package nl.tudelft.dnainator.javafx.drawables.annotations;

3   import javafx.beans.binding.DoubleBinding;
4   import javafx.scene.shape.Line;
5   import nl.tudelft.dnainator.javafx.drawables.strains.ClusterDrawable;
6   
7   /**
8    * The {@link Connection} is a {@link Line} that connects a {@link Gene} to a
9    * {@link ClusterDrawable}.
10   */
11  public class Connection extends Line {
12  
13  	/**
14  	 * Instantiates a new {@link Connection}.
15  	 * @param srcX The {@link DoubleBinding} to bind the start x-coordinate to.
16  	 * @param srcY The {@link DoubleBinding} to bind the start y-coordinate to.
17  	 * @param dstX The {@link DoubleBinding} to bind the end x-coordinate to.
18  	 * @param dstY The {@link DoubleBinding} to bind the end y-coordinate to.
19  	 */
20  	public Connection(DoubleBinding srcX, DoubleBinding srcY,
21  	                  DoubleBinding dstX, DoubleBinding dstY) {
22  		startXProperty().bind(srcX);
23  		startYProperty().bind(srcY);
24  		endXProperty().bind(dstX);
25  		endYProperty().bind(dstY);
26  	}
27  }
