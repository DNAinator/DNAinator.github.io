26  /**
27   * The {@link Strain} class represents the graph that contains the DNA strain.
28   * It holds both content and children, and toggles what to load and display based on the zoom level.
29   */
30  public class Strain extends SemanticDrawable {
31  	private static final int OFFSET = 10;
32  	private static final double ANNOTATION_HEIGHT = 50;
33  	private static final int RANK_WIDTH = 10;
34  	private static final int ZOOM_MIN = 200;
35  	private static final int ZOOM_MAX = 800;
36  
37  	private ColorServer colorServer;
38  	private Graph graph;
39  	private Map<String, ClusterDrawable> clusters;
40  	private IntegerProperty minRank = new SimpleIntegerProperty(0, "minRank");
41  	private IntegerProperty maxRank = new SimpleIntegerProperty(0, "maxRank");

43  	/**
44  	 * Construct a new top level {@link Strain} using the specified graph.
45  	 *
46  	 * @param colorServer The {@link ColorServer} to bind to.
47  	 * @param graph The specified graph.
48  	 */
49  	public Strain(ColorServer colorServer, Graph graph) {
50  		this(colorServer, graph, new Group(), new Group());

53  	/**
54  	 * Construct a new top level {@link Strain} using the specified graph, content and child
55  	 * content.
56  	 *
57  	 * @param colorServer  The {@link ColorServer} to bind to.
58  	 * @param graph        The specified graph.
59  	 * @param content      The specified graph content.
60  	 * @param childContent The specified child content.
61  	 */
62  	public Strain(ColorServer colorServer, Graph graph, Group content, Group childContent) {
63  		super(content, childContent);
64  		this.colorServer = colorServer;
65  		this.graph = graph;
66  		this.clusters = new HashMap<>();
67  	}

69  	@Override
70  	protected void loadContent(Bounds bounds) {
71  		content.getChildren().clear();
72  		HBox hbox = new HBox();
73  		Range ranks = getRange(bounds);
74  
75  		List<Annotation> annotations = getSortedAnnotations(ranks);
76  		annotations.forEach(a -> hbox.getChildren().add(new Gene(a)));
77  		content.getChildren().add(hbox);
78  	}
79  
80  	@Override
81  	public void loadChildren(Bounds bounds) {
82  		Range ranks = getRange(bounds);
83  		int interestingness = Math.max(0, Math.min((int) bounds.getWidth() - ZOOM_MIN, ZOOM_MAX));
84  		System.out.println("load iteration: " + ranks.getX() + " -> " + ranks.getY()
85  					+ " zoom: " + interestingness);
86  
87  		List<String> roots = graph.getRank(ranks.getX()).stream()
88  				.map(SequenceNode::getId).collect(Collectors.toList());
89  		List<Annotation> annotations = getSortedAnnotations(ranks);
90  		Map<Integer, List<Cluster>> result = graph.getAllClusters(roots, ranks.getY(),
91  				interestingness);
92  		clusters.clear();
93  		childContent.getChildren().clear();
94  
95  		result.forEach(this::loadRank);
96  		clusters.values().forEach(e -> loadEdges(bounds, e));
97  		drawAnnotations(annotations);
98  	}

100 	private Range getRange(Bounds bounds) {
101 		int min = (int) (Math.max(bounds.getMinX() / RANK_WIDTH, 0));
102 		int max = (int) (RANK_WIDTH + bounds.getMaxX() / RANK_WIDTH);
103 		min = Math.max(min, 0);
104 		min = Math.min(min, graph.getMaxRank());
105 		max = Math.max(min, max);
106 		max = Math.min(max, graph.getMaxRank());
107 		minRank.set(min);
108 		maxRank.set(max);
109 		return new Range(min, max);
110 	}
111 
112 	private List<Annotation> getSortedAnnotations(Range ranks) {
113 		return graph.getAnnotationByRank(ranks).stream()
114 				.sorted((a1, a2) -> Integer.compare(a1.getStart(), a2.getStart()))
115 				.collect(Collectors.toList());
116 	}
117 
118 	private void drawAnnotations(Collection<Annotation> annotations) {
119 		Gene prev = null;
120 		for (Annotation a : annotations) {
121 			prev = loadAnnotations(a, prev);
122 		}
123 	}
124 
125 	private Gene loadAnnotations(Annotation annotation, Gene prev) {
126 		Gene g = new Gene(annotation);
127 		ClusterDrawable left = getClusterDrawable(annotation, Double.MAX_VALUE,
128 				(x, acc) -> x <= acc);
129 		ClusterDrawable right = getClusterDrawable(annotation, Double.MIN_VALUE,
130 				(x, acc) -> x >= acc);
131 		if (left != null) {
132 			childContent.getChildren().add(new Connection(g.translateXProperty().add(0),
133 					g.translateYProperty().add(0), left.translateXProperty().add(0),
134 					left.translateYProperty().add(0)));
135 		}
136 		if (right != null) {
137 			childContent.getChildren().add(new Connection(g.translateXProperty().add(
138 					g.widthProperty()), g.translateYProperty().add(0),
139 					right.translateXProperty().add(0), right.translateYProperty().add(0)));
140 		}
141 		if (left != null && right != null) {
142 			g.translateXProperty().bind(Bindings.add(left.translateXProperty(), Bindings.subtract(
143 					Bindings.divide(right.translateXProperty().subtract(left.translateXProperty()),
144 							2), g.widthProperty().divide(2))));
145 			if (prev != null && g.getLayoutBounds().intersects(prev.getLayoutBounds())) {
146 				g.translateYProperty().bind(prev.translateYProperty().add(ANNOTATION_HEIGHT));
147 			} else {
148 				g.translateYProperty().bind(left.translateYProperty().add(ANNOTATION_HEIGHT));
149 			}
150 		}
151 		childContent.getChildren().add(g);
152 		return g;
153 	}

155 	private ClusterDrawable getClusterDrawable(Annotation annotation, double startValue,
156 	                                           BiFunction<Double, Double, Boolean> function) {
157 		Collection<String> ids = annotation.getAnnotatedNodes();
158 		ClusterDrawable res = null;
159 		double acc = startValue;
160 
161 		for (String id : ids) {
162 			ClusterDrawable cluster = clusters.get(id);
163 			if (cluster == null) {
164 				continue;
165 			}
166 			if (function.apply(cluster.getTranslateX(), acc)) {
167 				res = cluster;
168 				acc = cluster.getTranslateX();
169 			}
170 		}
171 		return res;
172 	}
173 
174 	private void loadEdges(Bounds bounds, ClusterDrawable cluster) {
175 		childContent.getChildren().addAll(cluster.getCluster().getNodes().stream()
176 				.flatMap(e -> e.getOutgoing().stream())
177 				.filter(destid -> clusters.get(destid) != cluster)
178 				.map(destid -> {
179 					ClusterDrawable dest = clusters.get(destid);
180 					if (dest == null) {
181 						Edge e = new Edge(cluster, destid);
182 						e.getEdge().endYProperty().setValue(bounds.getMinY() + OFFSET);
183 						return e;
184 					} else {
185 						return new Edge(cluster, dest);
186 					}
187 				})
188 				.collect(Collectors.toList()));
189 	}
190 
191 	private void loadRank(Integer key, List<Cluster> value) {
192 		for (int i = 0; i < value.size(); i++) {
193 			ClusterDrawable cluster = new ClusterDrawable(colorServer, value.get(i));
194 			cluster.getCluster().getNodes().forEach(e -> clusters.put(e.getId(), cluster));
195 			cluster.setTranslateX(key * RANK_WIDTH);
196 			cluster.setTranslateY(i * RANK_WIDTH - value.size() * RANK_WIDTH / 2);
197 			childContent.getChildren().add(cluster);
198 		}
199 	}
