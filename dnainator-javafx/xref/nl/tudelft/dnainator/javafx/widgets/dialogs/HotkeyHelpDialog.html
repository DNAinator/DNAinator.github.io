1   package nl.tudelft.dnainator.javafx.widgets.dialogs;
2   
3   import javafx.scene.Node;
4   import javafx.scene.control.Alert;
5   import javafx.scene.control.ButtonType;
6   import javafx.scene.control.Label;
7   import javafx.scene.control.Menu;
8   import javafx.scene.control.ButtonBar.ButtonData;
9   import javafx.scene.control.MenuItem;
10  import javafx.scene.layout.ColumnConstraints;
11  import javafx.scene.layout.GridPane;
12  import javafx.stage.Modality;
13  
14  /**
15   * Creates an {@link Alert} showing information regarding all the shortcuts of the controls.
16   */
17  public class HotkeyHelpDialog extends Alert {
18  	private Node parent;
19  	private static final String STYLE = "/style.css";
20  	private static final int COLUMN_WIDTH = 75;
21  	private static final int WIDTH = 300;
22  	private GridPane content;
23  	private int nextRow = 0;
24  
25  	/**
26  	 * Instantitates a new hotkey help dialog.
27  	 * @param parent   The parent Node of this dialog.
28  	 * @param menus    The menus carrying the shortcuts.
29  	 */
30  	public HotkeyHelpDialog(Node parent, Menu... menus) {
31  		super(Alert.AlertType.NONE);
32  		this.parent = parent;
33  		setupDialog(menus);
34  	}
35  	
36  	private void setupDialog(Menu... menus) {
37  		this.setTitle("Hotkeys");
38  		this.getDialogPane().getStylesheets().add(getClass().getResource(STYLE).toString());
39  		this.initOwner(parent.getScene().getWindow());
40  		this.initModality(Modality.NONE);
41  		this.getDialogPane().setPrefWidth(WIDTH);
42  		setupContentGrid();
43  		this.setHeaderText("Hotkeys");
44  		setupContent(menus);
45  
46  		getDialogPane().setContent(content);
47  		
48  		ButtonType close = new ButtonType("Close", ButtonData.CANCEL_CLOSE);
49  		this.getDialogPane().getButtonTypes().add(close);
50  		this.showingProperty().addListener((obj, oldV, newV) -> {
51  			MenuItem hotkey = menus[menus.length - 1].getItems().get(0);
52  			hotkey.setDisable(!hotkey.isDisable());
53  		});
54  	}
55  	
56  	private void setupContent(Menu... menus) {
57  		for (Menu menu : menus) {
58  			menu.getItems().forEach(menuItem -> {
59  				if (menuItem.getAccelerator() != null) {
60  					setupTitles(menuItem.getText());
61  					setupControls(menuItem.getAccelerator().getDisplayText());
62  				}
63  			});
64  		}
65  	}
66  	
67  	private void setupContentGrid() {
68  		content = new GridPane();
69  		ColumnConstraints col1 = new ColumnConstraints();
70  		col1.setPercentWidth(COLUMN_WIDTH);
71  		content.getColumnConstraints().add(col1);
72  	}
73  
74  	private void setupTitles(String title) {
75  		Label entry = new Label();
76  		entry.setText(title);
77  		entry.getStyleClass().add("hotkey-title");
78  		content.add(entry, 0, nextRow);
79  	}
80  	
81  	private void setupControls(String control) {
82  		Label controlEntry = new Label();
83  		controlEntry.setText(control);
84  		controlEntry.getStyleClass().add("hotkey-control");
85  		content.add(controlEntry, 1, nextRow++);
86  	}
87  }
