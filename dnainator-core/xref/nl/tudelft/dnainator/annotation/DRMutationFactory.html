6   import java.io.IOException;

8   /**
9    * This factory adds the known drug resistant mutations to the {@link AnnotationCollection}.
10   */
11  public class DRMutationFactory {
12  
13  	/**
14  	 * Parses and adds the known drug resistant mutations to the passed
15  	 * {@link AnnotationCollection}.
16  	 * @param annotations The annotations to which the {@link DRMutation}s are added.
17  	 * @param parser The parser that parses the drug resistant mutations file.
18  	 * @return The {@link AnnotationCollection} with the drug resistant mutations added to it.
19  	 */
20  	public AnnotationCollection build(AnnotationCollection annotations, Parser<DRMutation> parser) {
21  		try {
22  			while (parser.hasNext()) {
23  				DRMutation m = parser.next();
24  				annotations.addAnnotation(m);
25  			}
26  			parser.close();
27  		} catch (IOException e) {
28  			e.printStackTrace();
29  		}
30  		return annotations;
31  	}
32  }
