3   /**
4    * Interface for classes which can compile a
5    * {@link GraphQueryDescription} and {@link QueryElement}s.
6    * This is implemented according to the visitor pattern. The compile
7    * methods are the "visit" methods in the visitor pattern.
8    */
9   public interface GraphQuery {
10  
11  	/**
12  	 * Compiles the given {@link GraphQueryDescription}.
13  	 * @param qd The description for the query to be compiled.
14  	 */
15  	void compile(GraphQueryDescription qd);
16  
17  	/**
18  	 * @param ids The filter on node IDs to be compiled.
19  	 */
20  	void compile(IDsFilter ids);
21  
22  	/**
23  	 * @param sources The filter on sources to be compiled.
24  	 */
25  	void compile(SourcesFilter sources);
26  
27  	/**
28  	 * @param p The predicate to be compiled.
29  	 */
30  	void compile(PredicateFilter p);
31  
32  	/**
33  	 * @param start The start rank from where to start searching.
34  	 */
35  	void compile(RankStart start);
36  
37  	/**
38  	 * @param end The end rank where to stop searching.
39  	 */
40  	void compile(RankEnd end);
41  
42  }
