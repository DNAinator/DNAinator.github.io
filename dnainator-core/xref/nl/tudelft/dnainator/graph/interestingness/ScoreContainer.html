1   package nl.tudelft.dnainator.graph.interestingness;
2   
3   import java.util.Arrays;
4   import java.util.Map;
5   import java.util.function.Function;
6   import java.util.stream.Collectors;
7   
8   /**
9    * Represents a location within the graph.
10   */
11  public interface ScoreContainer {
12  	/**
13  	 * @param id the identifier of the property, which is used to get the score of
14  	 * a particular property.
15  	 * @return the score of a particular property
16  	 */
17  	int getScore(ScoreIdentifier id);
18  
19  	/**
20  	 * Returns all of the scores of this {@link ScoreContainer}. This uses getScore
21  	 * by default, but may be optimized by overriding it if the scores are stored
22  	 * internally as a {@link Map}.
23  	 * @return all of the scores of this {@link ScoreContainer}.
24  	 */
25  	default Map<ScoreIdentifier, Integer> getScores() {
26  		return Arrays.stream(Scores.values())
27  				.collect(Collectors.toMap(Function.identity(), this::getScore));
28  	}
29  }
