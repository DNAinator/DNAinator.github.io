1   package nl.tudelft.dnainator.graph;
2   
3   import nl.tudelft.dnainator.core.SequenceNode;
4   import nl.tudelft.dnainator.core.impl.Edge;
5   import nl.tudelft.dnainator.parser.Iterator;
6   
7   import java.io.IOException;
8   
9   /**
10   * Interface for constructing graphs.
11   */
12  public interface GraphBuilder {
13  
14  	/**
15  	 * Add an {@link Edge} to the graph.
16  	 * @param edge The edge containing the source and destination node.
17  	 * @return the {@link GraphBuilder}, after adding the edge.
18  	 */
19  	GraphBuilder addEdge(Edge<String> edge);
20  
21  	/**
22  	 * Add a {@link SequenceNode} to the graph.
23  	 * @param s The node to be added.
24  	 * @return the {@link GraphBuilder}, after adding the node.
25  	 */
26  	GraphBuilder addNode(SequenceNode s);
27  
28  	/**
29  	 * Construct a graph using the supplied parsers as a source for nodes and edges.
30  	 * @param np The {@link Iterator} to supply the nodes.
31  	 * @param ep The {@link Iterator} to supply the edges.
32  	 * @throws IOException If something goes wrong with IO while parsing.
33  	 * @return the {@link GraphBuilder}, after adding the nodes and edges.
34  	 */
35  	default GraphBuilder constructGraph(Iterator<SequenceNode> np, Iterator<Edge<String>> ep)
36  			throws IOException {
37  		try {
38  			while (np.hasNext()) {
39  				addNode(np.next());
40  			}
41  			while (ep.hasNext()) {
42  				addEdge(ep.next());
43  			}
44  		} finally {
45  			ep.close();
46  			np.close();
47  		}
48  		return this;
49  	}
50  
51  	/**
52  	 * Build the {@link Graph}.
53  	 * @return the constructed {@link Graph}.
54  	 */
55  	Graph build();
56  }
