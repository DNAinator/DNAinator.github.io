3   import nl.tudelft.dnainator.annotation.Annotation;

8   import org.neo4j.graphdb.Direction;
9   import org.neo4j.graphdb.GraphDatabaseService;
10  import org.neo4j.graphdb.Node;
11  import org.neo4j.graphdb.Transaction;
12  
13  import java.util.ArrayList;
14  import java.util.Collection;
15  
16  /**
17   * An {@link Annotation} which is stored in the Neo4j database.
18   */
19  public class Neo4jAnnotation implements Annotation {
20  	private String geneName;
21  	private Range range;
22  	private boolean isMutation;
23  	private boolean isSense;
24  	private Collection<String> annotatedNodes;
25  
26  	/**
27  	 * Constructs a new Neo4jAnnotation.
28  	 * @param service the service for accessing the database
29  	 * @param delegate the node to delegate to.
30  	 */
31  	public Neo4jAnnotation(GraphDatabaseService service, Node delegate) {
32  		try (Transaction tx = service.beginTx()) {
33  			this.geneName = (String) delegate.getProperty(AnnotationProperties.ID.name());
34  			this.range = new Range((int) delegate.getProperty(AnnotationProperties.STARTREF.name()),
35  					(Integer) delegate.getProperty(AnnotationProperties.ENDREF.name()));
36  			this.isMutation = delegate.hasLabel(NodeLabels.DRMUTATION);
37  			this.isSense = (Boolean) delegate.getProperty(AnnotationProperties.SENSE.name());
38  			this.annotatedNodes = new ArrayList<>();
39  			delegate.getRelationships(Direction.INCOMING, RelTypes.ANNOTATED).forEach(e ->
40  				annotatedNodes.add(
41  					(String) e.getStartNode().getProperty(SequenceProperties.ID.name())
42  				)
43  			);
44  			tx.success();
45  		}
46  	}
47  
48  	@Override
49  	public String getGeneName() {
50  		return geneName;
51  	}
52  
53  	@Override
54  	public Range getRange() {
55  		return range;
56  	}
57  
58  	@Override
59  	public boolean isMutation() {
60  		return isMutation;
61  	}
62  
63  	@Override
64  	public boolean isSense() {
65  		return isSense;
66  	}
67  
68  	@Override
69  	public Collection<String> getAnnotatedNodes() {
70  		return annotatedNodes;
71  	}
72  }
