1   package nl.tudelft.dnainator.graph.impl;
2   
3   import nl.tudelft.dnainator.core.SequenceNode;
4   import nl.tudelft.dnainator.core.impl.Edge;
5   import nl.tudelft.dnainator.graph.GraphBuilder;
6   import nl.tudelft.dnainator.parser.EdgeParser;
7   import nl.tudelft.dnainator.parser.NodeParser;
8   import nl.tudelft.dnainator.parser.exceptions.ParseException;
9   import org.neo4j.unsafe.batchinsert.BatchInserter;
10  import org.neo4j.unsafe.batchinsert.BatchInserters;
11  
12  import java.io.IOException;
13  import java.util.Collections;
14  import java.util.HashMap;
15  import java.util.Map;
16  
17  /**
18   * A {@link GraphBuilder} which uses a BatchInserter.
19   */
20  public class Neo4jBatchBuilder implements GraphBuilder {
21  	private BatchInserter batchInserter;
22  	private Map<String, Long> sequenceIDToNodeID;
23  	private Map<String, Long> sourceNodeIDs;
24  	private Map<String, Object> nodeProperties;
25  
26  	/**
27  	 * Create a new {@link Neo4jBatchBuilder}, which batch inserts to
28  	 * the specified database path.
29  	 * @param storeDir the path where the database is stored.
30  	 */
31  	public Neo4jBatchBuilder(String storeDir) {
32  		batchInserter = BatchInserters.inserter(storeDir);
33  		sequenceIDToNodeID = new HashMap<>();
34  		sourceNodeIDs = new HashMap<>();
35  		nodeProperties = new HashMap<>();
36  	}
37  
38  	@Override
39  	public void constructGraph(NodeParser np, EdgeParser ep)
40  			throws IOException, ParseException {
41  		GraphBuilder.super.constructGraph(np, ep);
42  		// Create the indices and constraints.
43  		batchInserter.createDeferredConstraint(NodeLabels.NODE)
44  			.assertPropertyIsUnique(PropertyTypes.ID.name())
45  			.create();
46  		batchInserter.createDeferredSchemaIndex(NodeLabels.NODE)
47  			.on(PropertyTypes.RANK.name())
48  			.create();
49  		batchInserter.createDeferredConstraint(NodeLabels.SOURCE)
50  			.assertPropertyIsUnique(PropertyTypes.SOURCE.name())
51  			.create();
52  		batchInserter.shutdown();
53  	}
54  
55  	@Override
56  	public void addEdge(Edge<String> edge) {
57  		batchInserter.createRelationship(sequenceIDToNodeID.get(edge.getSource()),
58  				sequenceIDToNodeID.get(edge.getDest()),
59  				RelTypes.NEXT, null);
60  	}
61  
62  	@Override
63  	public void addNode(SequenceNode s) {
64  		long nodeId = createNode(s);
65  		sequenceIDToNodeID.put(s.getId(), nodeId);
66  
67  		s.getSources().forEach(e -> {
68  			long source;
69  			if (!sourceNodeIDs.containsKey(e)) {
70  				source = createSource(e);
71  				sourceNodeIDs.put(e, source);
72  			} else {
73  				source = sourceNodeIDs.get(e);
74  			}
75  			batchInserter.createRelationship(nodeId, source, RelTypes.SOURCE, null);
76  		});
77  	}
78  
79  	private long createNode(SequenceNode s) {
80  		nodeProperties.put(PropertyTypes.ID.name(), s.getId());
81  		nodeProperties.put(PropertyTypes.STARTREF.name(), s.getStartRef());
82  		nodeProperties.put(PropertyTypes.ENDREF.name(), s.getEndRef());
83  		nodeProperties.put(PropertyTypes.SEQUENCE.name(), s.getSequence());
84  		nodeProperties.put(PropertyTypes.RANK.name(), 0);
85  		nodeProperties.put(PropertyTypes.SCORE.name(), s.getSequence().length());
86  		return batchInserter.createNode(nodeProperties, NodeLabels.NODE);
87  	}
88  
89  	private long createSource(String source) {
90  		return batchInserter.createNode(
91  				Collections.singletonMap(PropertyTypes.SOURCE.name(), source),
92  				NodeLabels.SOURCE);
93  	}
94  }
