28  		Mockito.when(container.getScore(Scores.DR_MUT)).thenReturn(TEST_DR_MUT);
29  	}
30  
31  	/**
32  	 * Tests whether it returns the sum of all scores.
33  	 */
34  	@Test
35  	public void testSum() {

37  				+ Scores.DR_MUT.applyImportanceModifier(TEST_DR_MUT);
