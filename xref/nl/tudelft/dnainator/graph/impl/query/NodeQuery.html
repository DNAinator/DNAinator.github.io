22  /**
23   * The {@link NodeQuery} returns one single {@link SequenceNode} using the specified neo4j node.
24   */
25  public class NodeQuery implements Query<SequenceNode> {
26  	private Node node;
27  
28  	/**
29  	 * Create a {@link NodeQuery} that will query for a Neo4j node.
30  	 * FIXME: Might change to String later?
31  	 * @param node	the specified Node
32  	 */
33  	public NodeQuery(Node node) {
34  		this.node = node;	
35  	}
36  	
37  	@Override
38  	public SequenceNode execute(GraphDatabaseService service) {
39  		String id	= (String) node.getProperty(ID.name());
40  		String source	= (String) node.getProperty(SOURCE.name());
41  		int startref	= (int)    node.getProperty(STARTREF.name());
42  		int endref	= (int)    node.getProperty(ENDREF.name());
43  		String sequence	= (String) node.getProperty(SEQUENCE.name());
44  		int rank	= (int)    node.getProperty(RANK.name());
45  
46  		List<String> outgoing = new ArrayList<>();
47  		for (Relationship e : loop(node.getRelationships(Direction.OUTGOING).iterator())) {
48  			outgoing.add((String) e.getEndNode().getProperty(ID.name()));
49  		}
50  
51  		return new SequenceNodeImpl(id, source, startref, endref, sequence, rank, outgoing);
52  	}
53  }
