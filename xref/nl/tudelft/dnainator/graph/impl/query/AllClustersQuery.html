6   import java.util.ArrayList;
7   import java.util.HashMap;
8   import java.util.HashSet;
9   import java.util.List;
10  import java.util.Map;
11  import java.util.PriorityQueue;
12  import java.util.Queue;
13  import java.util.Set;
14  
15  /**
16   * The {@link AllClustersQuery} creates {@link Cluster}s from all nodes,
17   * starting at the startNodes, and ending when the maximum specified start rank is reached.
18   */
19  public class AllClustersQuery implements Query<Map<Integer, List<Cluster>>> {
20  	private List<String> startNodes;
21  	private int threshold;
22  	private int maxRank;
23  
24  	/**
25  	 * Create a new {@link AllClustersQuery}, which will:.
26  	 * - start clustering at the specified startNodes
27  	 * - stop clustering when the end rank is reached
28  	 * - use the specified clustering threshold
29  	 * @param startNodes	the start nodes
30  	 * @param maxRank	the maximum rank
31  	 * @param threshold	the clustering threshold
32  	 */
33  	public AllClustersQuery(List<String> startNodes, int maxRank, int threshold) {
34  		this.startNodes = startNodes;
35  		this.maxRank = maxRank;
36  		this.threshold = threshold;
37  	}
38  
39  	@Override
40  	public Map<Integer, List<Cluster>> execute(GraphDatabaseService service) {
41  		Queue<Cluster> rootClusters = new PriorityQueue<>((e1, e2) -> 
42  			e1.getStartRank() - e2.getStartRank()
43  		);
44  		Set<String> visited = new HashSet<>();
45  		Map<Integer, List<Cluster>> result = new HashMap<Integer, List<Cluster>>();
46  	
47  		// Retrieve the root clusters starting from the startNodes and add them to the queue
48  		ClustersFromQuery clusterFrom = new ClustersFromQuery(visited, startNodes, threshold);
49  		rootClusters.addAll(clusterFrom.execute(service));
50  
51  		// Find adjacent clusters as long as there are root clusters in the queue
52  		int minrank = rootClusters.stream().mapToInt(e -> e.getStartRank()).min().orElse(0);
53  		while (!rootClusters.isEmpty()) {
54  			Cluster c = rootClusters.poll();
55  			if (c.getStartRank() < minrank || c.getStartRank() > maxRank) {
56  				continue;
57  			}
58  			result.putIfAbsent(c.getStartRank(), new ArrayList<>());
59  			result.get(c.getStartRank()).add(c);
60  
61  			c.getNodes().forEach(sn -> {
62  				ClustersFromQuery cfq = new ClustersFromQuery(visited,
63  										sn.getOutgoing(), threshold);
64  				rootClusters.addAll(cfq.execute(service));
65  			});
66  		}
67  
68  		return result;
69  	}
70  }
