21  /**
22   * The {@link ClusterQuery} creates one single {@link Cluster} from all nodes,
23   * starting at the start, that have a base sequence shorter than the specified threshold.
24   * It will only cluster nodes not yet marked as visited.
25   */
26  public class ClusterQuery implements Query<Cluster> {
27  	private Set<String> visited;
28  	private String start;
29  	private int threshold;
30  
31  	/**
32  	 * Create a new {@link ClusterQuery}, which will:.
33  	 * - only cluster nodes that haven't been visited yet
34  	 * - use the specified threshold
35  	 * - only return a single cluster
36  	 * @param visited	the visited nodes
37  	 * @param start		the start node
38  	 * @param threshold	the clustering threshold
39  	 */
40  	public ClusterQuery(Set<String> visited, String start, int threshold) {
41  		this.visited = visited;
42  		this.start = start;
43  		this.threshold = threshold;
44  	}
45  
46  	@Override
47  	public Cluster execute(GraphDatabaseService service) {
48  		Cluster cluster = null;
49  		Node startNode = service.findNode(NodeLabels.NODE, ID.name(), start);
50  		List<Node> result = new ArrayList<>();
51  
52  		// A depth first traversal traveling along both incoming and outgoing edges.
53  		TraversalDescription clusterDesc = service.traversalDescription()
54  						.depthFirst()
55  						.relationships(RelTypes.NEXT, Direction.BOTH)
56  						.evaluator(new ClusterEvaluator(threshold, visited));
57  		// Traverse the cluster starting from the startNode.
58  		int rankStart = (int) startNode.getProperty(RANK.name());
59  		for (Node end : clusterDesc.traverse(startNode).nodes()) {
60  			result.add(end);
61  
62  			// Update this cluster's start rank according to the lowest node rank.
63  			int endRank = (int) startNode.getProperty(RANK.name());
64  			if (endRank < rankStart) {
65  				rankStart = endRank;
66  			}
67  		}
68  
69  		// Might want to internally pass nodes.
70  		List<SequenceNode> retrieve = result.stream().map(e -> new Neo4jSequenceNode(service, e))
71  						.collect(Collectors.toList());
72  		cluster = new Cluster(rankStart, retrieve);
73  
74  		return cluster;
75  	}
76  }
