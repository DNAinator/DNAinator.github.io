12  import org.neo4j.unsafe.batchinsert.BatchInserter;
13  import org.neo4j.unsafe.batchinsert.BatchInserters;
14  
15  import java.util.Collections;
16  import java.util.HashMap;
17  import java.util.Map;

19  /**
20   * A {@link GraphBuilder} which uses a BatchInserter.
21   */
22  public class Neo4jBatchBuilder implements GraphBuilder {
23  	private String storeDir;
24  	private BatchInserter batchInserter;
25  	private Map<String, Long> annotationIDToNodeID;
26  	private Map<String, Long> sequenceIDToNodeID;
27  	private Map<String, Long> sourceToNodeID;
28  	private Map<String, Object> annotationProperties;
29  	private Map<String, Object> nodeProperties;
30  	private AnnotationCollection annotations;
31  	private TreeNode phylogeny;
32  
33  	/**
34  	 * Create a new {@link Neo4jBatchBuilder}, which batch inserts to
35  	 * the specified database path.
36  	 * @param storeDir the path where the database is stored.
37  	 * @param annotations the {@link AnnotationCollection} to connect the nodes with.
38  	 */
39  	public Neo4jBatchBuilder(String storeDir, AnnotationCollection annotations) {
40  		this(storeDir, annotations, null);
41  	}
42  
43  	/**
44  	 * Create a new {@link Neo4jBatchBuilder}, which batch inserts to
45  	 * the specified database path.
46  	 * @param storeDir the path where the database is stored.
47  	 * @param annotations the {@link AnnotationCollection} to connect the nodes with.
48  	 * @param phylogeny the phylogenetic tree, for linking the source nodes together.
49  	 */
50  	public Neo4jBatchBuilder(String storeDir, AnnotationCollection annotations,
51  			TreeNode phylogeny) {
52  		this.annotations = annotations;
53  		this.storeDir = storeDir;
54  		this.phylogeny = phylogeny;
55  
56  		batchInserter = BatchInserters.inserter(storeDir);
57  		annotationIDToNodeID = new HashMap<>();
58  		sequenceIDToNodeID = new HashMap<>();
59  		sourceToNodeID = new HashMap<>();
60  		annotationProperties = new HashMap<>();
61  		nodeProperties = new HashMap<>();
62  
63  		annotations.getAll().forEach(e -> {
64  			annotationIDToNodeID.put(e.getGeneName(), createAnnotation(e));
65  		});
66  	}
67  
68  	@Override
69  	public GraphBuilder addEdge(Edge<String> edge) {
70  		batchInserter.createRelationship(sequenceIDToNodeID.get(edge.getSource()),
71  				sequenceIDToNodeID.get(edge.getDest()),
72  				RelTypes.NEXT, null);
73  		return this;
74  	}
75  
76  	@Override
77  	public GraphBuilder addNode(SequenceNode s) {
78  		long nodeId = createNode(s);
79  		sequenceIDToNodeID.put(s.getId(), nodeId);
80  
81  		s.getSources().forEach(source -> {
82  			connectSource(nodeId, source);
83  			if (source.equals("TKK_REF")) {
84  				annotations.getSubrange(s.getStartRef(), s.getEndRef())
85  					.forEach(a -> connectAnnotation(nodeId, a));
86  			}
87  		});
88  		return this;
89  	}
90  
91  	@Override
92  	public Graph build() {
93  		connectPhylogeny();
94  		createIndicesAndConstraints();
95  		batchInserter.shutdown();
96  
97  		// Initialize the graph.
98  		Neo4jGraph g = new Neo4jGraph(storeDir);
99  		g.analyze();
100 		return g;
101 	}
102 
103 	/**
104 	 * Create all indices using deferred initialization.
105 	 */
106 	private void createIndicesAndConstraints() {
107 		batchInserter.createDeferredConstraint(NodeLabels.NODE)
108 			.assertPropertyIsUnique(PropertyTypes.ID.name())
109 			.create();
110 		batchInserter.createDeferredConstraint(NodeLabels.SOURCE)
111 			.assertPropertyIsUnique(PropertyTypes.SOURCE.name())
112 			.create();
113 		batchInserter.createDeferredSchemaIndex(NodeLabels.NODE)
114 			.on(PropertyTypes.RANK.name())
115 			.create();
116 		batchInserter.createDeferredSchemaIndex(NodeLabels.NODE)
117 			.on(PropertyTypes.STARTREF.name())
118 			.create();
119 		batchInserter.createDeferredSchemaIndex(NodeLabels.NODE)
120 			.on(PropertyTypes.ENDREF.name())
121 			.create();
122 	}
123 
124 	/**
125 	 * Connect a sequence node to an annotation node.
126 	 * @param nodeId	the sequence node
127 	 * @param source	the annotation
128 	 */
129 	private void connectAnnotation(long nodeId, Annotation annotation) {
130 		long annotationId = annotationIDToNodeID.get(annotation.getGeneName());
131 		batchInserter.createRelationship(nodeId, annotationId, RelTypes.ANNOTATED, null);
132 	}
133 
134 	/**
135 	 * Connect a sequence node to a source.
136 	 * @param nodeId	the sequence node
137 	 * @param source	the source node
138 	 */
139 	private void connectSource(long nodeId, String source) {
140 		long sourceId;
141 		if (!sourceToNodeID.containsKey(source)) {
142 			sourceId = createSource(source);
143 			sourceToNodeID.put(source, sourceId);
144 		} else {
145 			sourceId = sourceToNodeID.get(source);
146 		}
147 		batchInserter.createRelationship(nodeId, sourceId, RelTypes.SOURCE, null);
148 	}
149 
150 	/**
151 	 * Connect the root phylogeny node and all of its children to their sources (recursively).
152 	 */
153 	private void connectPhylogeny() {
154 		if (phylogeny != null) {
155 			connectPhylogeny(0, phylogeny);
156 		}
157 	}
158 
159 	/**
160 	 * Connect a phylogeny node and its children to their sources.
161 	 * @param distanceToRoot	distance of this node to root of the tree
162 	 * @param current		the current phylogeny tree node
163 	 * @return			the id of the created node
164 	 */
165 	private long connectPhylogeny(int distanceToRoot, TreeNode current) {
166 		if (current.getChildren().size() == 0) {
167 			return sourceToNodeID.get(current.getName());
168 		}
169 		long anc = createAncestor(distanceToRoot);
170 		current.getChildren().forEach(child ->
171 			batchInserter.createRelationship(anc,
172 					connectPhylogeny(distanceToRoot + 1, child),
173 					RelTypes.ANCESTOR_OF, null)
174 		);
175 		return anc;
176 	}
177 
178 	/**
179 	 * Create an annotation node.
180 	 * @param a	the annotation
181 	 * @return	the id of the created node
182 	 */
183 	private long createAnnotation(Annotation a) {
184 		annotationProperties.put(PropertyTypes.ID.name(), a.getGeneName());
185 		annotationProperties.put(PropertyTypes.STARTREF.name(), a.getStart());
186 		annotationProperties.put(PropertyTypes.ENDREF.name(), a.getEnd());
187 		annotationProperties.put(PropertyTypes.SENSE.name(), a.isSense());
188 		return batchInserter.createNode(annotationProperties, NodeLabels.ANNOTATION);
189 	}
190 
191 	/**
192 	 * Create a phylogeny ancestor node.
193 	 * @param distanceToRoot	distance of this node to root of the tree
194 	 * @return			the id of the created node
195 	 */
196 	private long createAncestor(int distanceToRoot) {
197 		return batchInserter.createNode(
198 				Collections.singletonMap(PropertyTypes.DIST_TO_ROOT.name(), distanceToRoot),
199 				NodeLabels.ANCESTOR);
200 	}
201 
202 	/**
203 	 * Create a sequence node.
204 	 * @param s	the sequencenode
205 	 * @return	the id of the created node
206 	 */
207 	private long createNode(SequenceNode s) {
208 		nodeProperties.put(PropertyTypes.ID.name(), s.getId());
209 		nodeProperties.put(PropertyTypes.STARTREF.name(), s.getStartRef());
210 		nodeProperties.put(PropertyTypes.ENDREF.name(), s.getEndRef());
211 		nodeProperties.put(PropertyTypes.SEQUENCE.name(), s.getSequence());
212 		nodeProperties.put(PropertyTypes.RANK.name(), 0);
213 		nodeProperties.put(Scores.SEQ_LENGTH.getName(), s.getSequence().length());
214 		return batchInserter.createNode(nodeProperties, NodeLabels.NODE);
215 	}
216 
217 	/**
218 	 * Create a source node.
219 	 * @param s	the source
220 	 * @return	the id of the created node
221 	 */
222 	private long createSource(String source) {
223 		return batchInserter.createNode(
224 				Collections.singletonMap(PropertyTypes.SOURCE.name(), source),
225 				NodeLabels.SOURCE);
226 	}
