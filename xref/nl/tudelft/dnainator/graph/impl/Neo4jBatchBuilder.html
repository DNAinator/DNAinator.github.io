1   package nl.tudelft.dnainator.graph.impl;
2   
3   import java.io.IOException;
4   import java.util.Collections;
5   import java.util.HashMap;
6   import java.util.Map;
7   
8   import org.neo4j.graphdb.DynamicLabel;
9   import org.neo4j.graphdb.Label;
10  import org.neo4j.unsafe.batchinsert.BatchInserter;
11  import org.neo4j.unsafe.batchinsert.BatchInserters;
12  
13  import nl.tudelft.dnainator.core.SequenceNode;
14  import nl.tudelft.dnainator.core.impl.Edge;
15  import nl.tudelft.dnainator.graph.GraphBuilder;
16  import nl.tudelft.dnainator.parser.EdgeParser;
17  import nl.tudelft.dnainator.parser.NodeParser;
18  import nl.tudelft.dnainator.parser.exceptions.ParseException;
19  
20  /**
21   * A {@link GraphBuilder} which uses a BatchInserter.
22   */
23  public class Neo4jBatchBuilder implements GraphBuilder {
24  	private BatchInserter batchInserter;
25  	private Map<String, Long> sequenceIDToNodeID;
26  	private Map<String, Long> sourceNodeIDs;
27  	private Map<String, Object> nodeProperties;
28  	private Label nodeLabel;
29  	private Label sourceLabel;
30  
31  	/**
32  	 * Create a new {@link Neo4jBatchBuilder}, which batch inserts to
33  	 * the specified database path.
34  	 * @param storeDir the path where the database is stored.
35  	 */
36  	public Neo4jBatchBuilder(String storeDir) {
37  		batchInserter = BatchInserters.inserter(storeDir);
38  		sequenceIDToNodeID = new HashMap<>();
39  		sourceNodeIDs = new HashMap<>();
40  		nodeProperties = new HashMap<>();
41  		// Assign a label to our nodes
42  		nodeLabel = DynamicLabel.label(PropertyTypes.NODELABEL.name());
43  		sourceLabel = DynamicLabel.label(PropertyTypes.SOURCE.name());
44  	}
45  
46  	@Override
47  	public void constructGraph(NodeParser np, EdgeParser ep)
48  			throws IOException, ParseException {
49  		GraphBuilder.super.constructGraph(np, ep);
50  		// Create the indices and constraints.
51  		batchInserter.createDeferredConstraint(nodeLabel)
52  			.assertPropertyIsUnique(PropertyTypes.ID.name())
53  			.create();
54  		batchInserter.createDeferredSchemaIndex(nodeLabel)
55  			.on(PropertyTypes.RANK.name())
56  			.create();
57  		batchInserter.createDeferredConstraint(sourceLabel)
58  			.assertPropertyIsUnique(PropertyTypes.SOURCE.name())
59  			.create();
60  		batchInserter.shutdown();
61  	}
62  
63  	@Override
64  	public void addEdge(Edge<String> edge) {
65  		batchInserter.createRelationship(sequenceIDToNodeID.get(edge.getSource()),
66  				sequenceIDToNodeID.get(edge.getDest()),
67  				RelTypes.NEXT, null);
68  	}
69  
70  	@Override
71  	public void addNode(SequenceNode s) {
72  		long nodeId = createNode(s);
73  		sequenceIDToNodeID.put(s.getId(), nodeId);
74  
75  		s.getSources().forEach(e -> {
76  			long source;
77  			if (!sourceNodeIDs.containsKey(e)) {
78  				source = createSource(e);
79  				sourceNodeIDs.put(e, source);
80  			} else {
81  				source = sourceNodeIDs.get(e);
82  			}
83  			batchInserter.createRelationship(nodeId, source, RelTypes.SOURCE, null);
84  		});
85  	}
86  
87  	private long createNode(SequenceNode s) {
88  		nodeProperties.put(PropertyTypes.ID.name(), s.getId());
89  		nodeProperties.put(PropertyTypes.STARTREF.name(), s.getStartRef());
90  		nodeProperties.put(PropertyTypes.ENDREF.name(), s.getEndRef());
91  		nodeProperties.put(PropertyTypes.SEQUENCE.name(), s.getSequence());
92  		nodeProperties.put(PropertyTypes.RANK.name(), 0);
93  		nodeProperties.put(PropertyTypes.SCORE.name(), s.getSequence().length());
94  		return batchInserter.createNode(nodeProperties, nodeLabel);
95  	}
96  
97  	private long createSource(String source) {
98  		return batchInserter.createNode(
99  				Collections.singletonMap(PropertyTypes.SOURCE.name(), source),
100 				sourceLabel);
101 	}
102 }
