1   package nl.tudelft.dnainator.graph.impl.command;
2   
3   import static nl.tudelft.dnainator.graph.impl.PropertyTypes.RANK;
4   import static org.neo4j.helpers.collection.IteratorUtil.loop;
5   import nl.tudelft.dnainator.graph.impl.RelTypes;
6   
7   import org.neo4j.collection.primitive.Primitive;
8   import org.neo4j.collection.primitive.PrimitiveLongSet;
9   import org.neo4j.graphdb.Direction;
10  import org.neo4j.graphdb.GraphDatabaseService;
11  import org.neo4j.graphdb.Node;
12  import org.neo4j.graphdb.Relationship;
13  import org.neo4j.graphdb.ResourceIterator;
14  import org.neo4j.graphdb.Transaction;
15  import org.neo4j.graphdb.traversal.InitialBranchState.State;
16  import org.neo4j.graphdb.traversal.Uniqueness;
17  
18  /**
19   * The {@link RankCommand} creates a topological ordering and
20   * ranks the nodes in the Neo4j database accordingly.
21   */
22  public class RankCommand implements Command {
23  	private static final int INIT_CAP = 4096;
24  	private ResourceIterator<Node> roots;
25  
26  	/**
27  	 * Create a new {@link RankCommand} that will
28  	 * start ranking from the specified roots.
29  	 * @param roots	the roots
30  	 */
31  	public RankCommand(ResourceIterator<Node> roots) {
32  		this.roots = roots;
33  	}
34  
35  	/**
36  	 * Return a topological ordering on the specified database service.
37  	 * @param service	the database service
38  	 * @return		a topological ordering, starting from the roots
39  	 */
40  	public Iterable<Node> topologicalOrder(GraphDatabaseService service) {
41  		return topologicalOrder(service, Primitive.longSet());
42  	}
43  
44  	private Iterable<Node> topologicalOrder(GraphDatabaseService service,
45  						PrimitiveLongSet processed) {
46  		return service.traversalDescription()
47  				.depthFirst()
48  				.expand(new TopologicalPathExpander()
49  				, new State<>(processed, null))
50  				// We manage uniqueness for ourselves.
51  				.uniqueness(Uniqueness.NONE)
52  				.traverse(loop(roots))
53  				.nodes();
54  	}
55  
56  	@Override
57  	public void execute(GraphDatabaseService service) {
58  		try (
59  			Transaction tx = service.beginTx();
60  			// Our set is located "off heap", i.e. not managed by the garbage collector.
61  			// It is automatically closed after the try block, which frees the allocated memory.
62  			PrimitiveLongSet processed = Primitive.offHeapLongSet(INIT_CAP)
63  		) {
64  			for (Node n : topologicalOrder(service, processed)) {
65  				int rankSource = (int) n.getProperty(RANK.name());
66  				for (Relationship r : n.getRelationships(RelTypes.NEXT, Direction.OUTGOING)) {
67  					Node dest = r.getEndNode();
68  					if ((int) dest.getProperty(RANK.name()) < rankSource + 1) {
69  						dest.setProperty(RANK.name(), rankSource + 1);
70  					}
71  				}
72  			}
73  			tx.success();
74  		}
75  	}
76  }
