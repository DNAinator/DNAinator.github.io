22  /**
23   * The {@link AnalyzeCommand} creates a topological ordering and
24   * ranks the nodes in the Neo4j database accordingly.
25   */
26  public class AnalyzeCommand implements Command {
27  	private static final int INIT_CAP = 4096;
28  	private ResourceIterator<Node> roots;
29  
30  	/**
31  	 * Create a new {@link AnalyzeCommand} that will
32  	 * start ranking from the specified roots.
33  	 * @param roots	the roots
34  	 */
35  	public AnalyzeCommand(ResourceIterator<Node> roots) {
36  		this.roots = roots;
37  	}
38  
39  	/**
40  	 * Return a topological ordering on the specified database service.
41  	 * @param service	the database service
42  	 * @return		a topological ordering, starting from the roots
43  	 */
44  	public Iterable<Node> topologicalOrder(GraphDatabaseService service) {
45  		return topologicalOrder(service, Primitive.longSet());
46  	}
47  
48  	private Iterable<Node> topologicalOrder(GraphDatabaseService service,
49  						PrimitiveLongSet processed) {
50  		return service.traversalDescription()
51  				.depthFirst()
52  				.expand(new TopologicalPathExpander()
53  				, new State<>(processed, null))
54  				// We manage uniqueness for ourselves.
55  				.uniqueness(Uniqueness.NONE)
56  				.traverse(loop(roots))
57  				.nodes();
58  	}
59  
60  	@Override
61  	public void execute(GraphDatabaseService service) {
62  		try (
63  			Transaction tx = service.beginTx();
64  			// Our set is located "off heap", i.e. not managed by the garbage collector.
65  			// It is automatically closed after the try block, which frees the allocated memory.
66  			PrimitiveLongSet processed = Primitive.offHeapLongSet(INIT_CAP)
67  		) {
68  			for (Node n : topologicalOrder(service, processed)) {
69  				rankDest(n);
70  			}
71  			tx.success();
72  		}
73  	}
74  
75  	private void rankDest(Node n) {
76  		int baseSource = (int) n.getProperty(BASE_DIST.name())
77  				+ (int) n.getProperty(Scores.SEQ_LENGTH.name());
78  		int rankSource = (int) n.getProperty(RANK.name()) + 1;
79  
80  		for (Relationship r : n.getRelationships(RelTypes.NEXT, Direction.OUTGOING)) {
81  			Node dest = r.getEndNode();
82  
83  			if ((int) dest.getProperty(BASE_DIST.name()) < baseSource) {
84  				dest.setProperty(BASE_DIST.name(), baseSource);
85  			}
86  			if ((int) dest.getProperty(RANK.name()) < rankSource) {
87  				dest.setProperty(RANK.name(), rankSource);
88  			}
89  		}
90  	}
91  }
