1   package nl.tudelft.dnainator.graph;
2   
3   import java.util.List;
4   import java.util.Map;
5   import java.util.Queue;
6   import java.util.Set;
7   
8   import nl.tudelft.dnainator.core.SequenceNode;
9   import nl.tudelft.dnainator.core.impl.Cluster;
10  import nl.tudelft.dnainator.graph.query.GraphQueryDescription;
11  
12  /**
13   * Interface for backend agnostic interaction with a graph.
14   */
15  public interface Graph {
16  	/**
17  	 * Get the root node of this graph.
18  	 * FIXME: This is the node that has no incoming edges.
19  	 * @return	a SequenceNode
20  	 */
21  	SequenceNode getRootNode();
22  
23  	/**
24  	 * Get the node with identifier n from this graph.
25  	 * @param n	the identifier
26  	 * @return	a SequenceNode
27  	 */
28  	SequenceNode getNode(String n);
29  
30  	/**
31  	 * Get all the nodes with a specific rank from this graph.
32  	 * @param rank	the rank
33  	 * @return		a list of sequence nodes
34  	 */
35  	List<SequenceNode> getRank(int rank);
36  
37  	/**
38  	 * Get a list of all nodes from this graph.
39  	 * @return	a list of all nodes, per rank
40  	 */
41  	List<List<SequenceNode>> getRanks();
42  
43  	/**
44  	 * Return a list of nodes that belong to the same cluster as the given startId.
45  	 * @param visited	nodes that were already visited
46  	 * @param startNodes	the start nodes
47  	 * @param threshold		the clustering threshold
48  	 * @return		a list representing the cluster
49  	 */
50  	Queue<Cluster> getClustersFrom(Set<String> visited, List<String> startNodes, int threshold);
51  
52  	/**
53  	 * Return a list of nodes that belong to the same cluster as the given startId.
54  	 * @param startNodes	the start nodes
55  	 * @param end		the maximum rank of the cluster
56  	 * @param threshold	the clustering threshold
57  	 * @return		a list representing the cluster
58  	 */
59  	Map<Integer, List<Cluster>> getAllClusters(List<String> startNodes, int end, int threshold);
60  
61  	/**
62  	 * Find the nodes satisfying the given query.
63  	 * @param q the query for finding the nodes.
64  	 * @return the result of the query.
65  	 */
66  	List<SequenceNode> queryNodes(GraphQueryDescription q);
67  }
