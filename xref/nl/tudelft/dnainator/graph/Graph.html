5   import nl.tudelft.dnainator.annotation.Range;
6   import nl.tudelft.dnainator.core.EnrichedSequenceNode;
7   import nl.tudelft.dnainator.core.impl.Cluster;
8   import nl.tudelft.dnainator.graph.interestingness.InterestingnessStrategy;

11  import java.util.Collection;
12  import java.util.List;
13  import java.util.Map;
14  
15  /**
16   * Interface for backend agnostic interaction with a graph.
17   */
18  public interface Graph extends AnnotationCollection {
19  	/**
20  	 * Get the root node of this graph.
21  	 * FIXME: This is the node that has no incoming edges.
22  	 * @return	a SequenceNode
23  	 */
24  	EnrichedSequenceNode getRootNode();
25  
26  	/**
27  	 * Get the node with identifier n from this graph.
28  	 * @param n	the identifier
29  	 * @return	a SequenceNode
30  	 */
31  	EnrichedSequenceNode getNode(String n);
32  
33  	/**
34  	 * @return The {@link AnnotationCollection} containing the annotations.
35  	 */
36  	AnnotationCollection getAnnotations();
37  
38  	/**
39  	 * Get all the nodes with a specific rank from this graph.
40  	 * @param rank	the rank
41  	 * @return		a list of sequence nodes
42  	 */
43  	List<EnrichedSequenceNode> getRank(int rank);
44  
45  	/**
46  	 * Return the maximum rank in this graph.
47  	 * @return	the maximum rank
48  	 */
49  	int getMaxRank();
50  
51  	/**
52  	 * Return a list of nodes that belong to the same cluster as the given startId.
53  	 * @param startNodes	the start nodes
54  	 * @param end		the maximum rank of the cluster
55  	 * @param threshold	the clustering threshold
56  	 * @return		a list representing the cluster
57  	 */
58  	Map<Integer, List<Cluster>> getAllClusters(List<String> startNodes, int end, int threshold);
59  
60  	/**
61  	 * Sets the interestingness strategy which calculates the interestingness when
62  	 * clustering.
63  	 * @param is the interestingness strategy.
64  	 */
65  	void setInterestingnessStrategy(InterestingnessStrategy is);
66  
67  	/**
68  	 * Find the nodes satisfying the given query.
69  	 * @param q the query for finding the nodes.
70  	 * @return the result of the query.
71  	 */
72  	List<EnrichedSequenceNode> queryNodes(GraphQueryDescription q);
73  
74  	/**
75  	 * Return all annotations covered by the given range of ranks.
76  	 * @param r The range
77  	 * @return all annotations covered.
78  	 */
79  	Collection<Annotation> getAnnotationByRank(Range r);
80  }
