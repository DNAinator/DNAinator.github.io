36   * This class realizes a graphfactory using Neo4j as it's backend.

38  public final class Neo4jGraphDatabase implements Graph {
39  	private static final String GET_ROOT = "MATCH (s:Node) "
40  			+ "WHERE NOT (s)<-[:NEXT]-(:Node)"
41  			+ "RETURN s";

53  	/**
54  	 * Constructs a Neo4j database on the specified path.
55  	 * @param path			specified path
56  	 */
57  	public Neo4jGraphDatabase(String path) {
58  		this(new GraphDatabaseFactory().newEmbeddedDatabase(path));

61  	/**
62  	 * Constructs a Neo4j database on the specified path.
63  	 * This constructor is primarily intended to be used from JUnit.
64  	 * @param service		specified path
65  	 */
66  	protected Neo4jGraphDatabase(GraphDatabaseService service) {
67  		// Create our database and register a shutdown hook
68  		this.service = service;
69  		Runtime.getRuntime().addShutdownHook(new Thread() {
70  			@Override
71  			public void run() {
72  				service.shutdown();
73  			}
74  		});
75  
76  		// Assign a label to our nodes
77  		nodeLabel = DynamicLabel.label("Node");
78  		try (Transaction tx = service.beginTx()) {
79  			// Generate a unique index on 'id'
80  			service.schema().constraintFor(nodeLabel)
81  			.assertPropertyIsUnique("id")
82  			.create();
83  
84  			// Generate an index on 'dist'
85  			service.schema().indexFor(nodeLabel)
86  			.on("dist")
87  			.create();
88  
89  			tx.success();
90  		}
91  	}

93  	@Override
94  	public void addEdge(Edge<String> edge) {
95  		try (Transaction tx = service.beginTx()) {
96  			Node source = service.findNode(nodeLabel, "id", edge.getSource());
97  			Node dest   = service.findNode(nodeLabel, "id", edge.getDest());
98  			source.createRelationshipTo(dest, RelTypes.NEXT);
99  
100 			tx.success();
101 		}
102 	}

104 	@Override
105 	public void addNode(SequenceNode s) {
106 		try (Transaction tx = service.beginTx()) {
107 			Node node = service.createNode(nodeLabel);
108 			node.setProperty("id", s.getId());
109 			node.setProperty("start", s.getStartRef());
110 			node.setProperty("end", s.getEndRef());
111 			node.setProperty("sequence", s.getSequence());
112 			node.setProperty("source", s.getSource());
113 			node.setProperty("dist", 0);
114 
115 			tx.success();
116 		}
117 	}

119 	@Override
120 	public void constructGraph(NodeParser np, EdgeParser ep)
121 			throws IOException, ParseException {
122 		try (Transaction tx = service.beginTx()) {
123 			while (np.hasNext()) {
124 				addNode(np.next());
125 			}
126 			while (ep.hasNext()) {
127 				addEdge(ep.next());
128 			}
129 			rankGraph();
130 
131 			tx.success();
132 		}
133 	}

135 	/**
136 	 * Retrieves a resource iterator over all roots.
137 	 * The roots are all the nodes with no incoming edges.
138 	 * @return	a resource iterator
139 	 */
140 	private ResourceIterator<Node> rootIterator() {
141 		ResourceIterator<Node> roots;
142 		Result res = service.execute(GET_ROOT);
143 		roots = res.columnAs("s");
144 		return roots;
145 	}
146 
147 	private Node getRoot() {
148 		return rootIterator().next();
149 	}

151 	private static final int INIT_CAP = 4096;
152 	/**
153 	 * Get a topological ordering of the graph.
154 	 * @return an {@link Iterable}, containing the nodes in
155 	 * topological order.
156 	 */
157 	protected Iterable<Node> topologicalOrder() {
158 		return topologicalOrder(Primitive.longSet(INIT_CAP));
159 	}
160 
161 	private Iterable<Node> topologicalOrder(PrimitiveLongSet processed) {
162 		ResourceIterator<Node> roots = rootIterator();
163 		return service.traversalDescription()
164 					.depthFirst()
165 					.expand(new IncludesNodesWithoutIncoming()
166 					, new State<PrimitiveLongSet>(processed, null))
167 					// We manage uniqueness for ourselves.
168 					.uniqueness(Uniqueness.NONE)
169 					.traverse(loop(roots))
170 					.nodes();
171 	}
172 
173 	/**
174 	 * PathExpander for determining the topological ordering.
175 	 */
176 	static class IncludesNodesWithoutIncoming implements PathExpander<PrimitiveLongSet> {
177 
178 		private boolean hasUnprocessedIncoming(PrimitiveLongSet processed, Node n) {
179 			Iterable<Relationship> in = n.getRelationships(RelTypes.NEXT, Direction.INCOMING);
180 			for (Relationship r : in) {
181 				if (!processed.contains(r.getId())) {
182 					return true;
183 				}
184 			}
185 			// All incoming edges have been processed.
186 			return false;
187 		}
188 
189 		@Override
190 		public Iterable<Relationship> expand(Path path,
191 				BranchState<PrimitiveLongSet> state) {
192 			Node from = path.endNode();
193 			List<Relationship> expand = new LinkedList<Relationship>();
194 			for (Relationship r : from.getRelationships(RelTypes.NEXT, Direction.OUTGOING)) {
195 				PrimitiveLongSet processed = state.getState();
196 				processed.add(r.getId());
197 				if (!hasUnprocessedIncoming(processed, r.getEndNode())) {
198 					// All of the dependencies of this node have been added to the result.
199 					expand.add(r);
200 				}
201 			}
202 			return expand;
203 		}
204 
205 		@Override
206 		public PathExpander<PrimitiveLongSet> reverse() {
207 			throw new UnsupportedOperationException();
208 		}
209 
210 	}

212 	private void rankGraph() {
213 		try (
214 			Transaction tx = service.beginTx();
215 			// Our set is located "off heap", i.e. not managed by the garbage collector.
216 			// It is automatically closed after the try block, which frees the allocated memory.
217 			PrimitiveLongSet processed = Primitive.offHeapLongSet(INIT_CAP)
218 		) {
219 			for (Node n : topologicalOrder(processed)) {
220 				int rankSource = (int) n.getProperty("dist");
221 				for (Relationship r : n.getRelationships(RelTypes.NEXT, Direction.OUTGOING)) {
222 					Node dest = r.getEndNode();
223 					if ((int) dest.getProperty("dist") < rankSource + 1) {
224 						dest.setProperty("dist", rankSource + 1);
225 					}
226 				}
227 			}
228 			tx.success();
229 		}
230 	}

247 	public SequenceNode getNode(String s) {
248 		SequenceNode node = null;
249 
250 		try (Transaction tx = service.beginTx()) {
251 			node = createSequenceNode(service.findNode(nodeLabel, "id", s));
252 
253 			tx.success();
254 		}
255 
256 		return node;

259 	@Override
260 	public List<SequenceNode> getRank(int rank) {
261 		List<SequenceNode> nodes = new LinkedList<SequenceNode>();
262 
263 		try (Transaction tx = service.beginTx()) {
264 			ResourceIterator<Node> res = service.findNodes(nodeLabel, "dist", rank);
265 
266 			for (Node n : loop(res)) {
267 				nodes.add(createSequenceNode(n));
268 			}
269 
270 			tx.success();
271 		}
272 
273 		return nodes;
274 	}

276 	private SequenceNode createSequenceNode(Node node) {
277 		String id       = (String) node.getProperty("id");
278 		String source   = (String) node.getProperty("source");
279 		int startref    = (int) node.getProperty("start");
280 		int endref      = (int) node.getProperty("end");
281 		String sequence = (String) node.getProperty("sequence");
282 
283 		return new SequenceNodeImpl(id, source, startref, endref, sequence);
284 	}
