4   import nl.tudelft.dnainator.util.Edge;

6   import java.io.IOException;
7   import java.util.NoSuchElementException;
8   
9   /**
10   * An interface for parsing edge entries, using the Iterator pattern.
11   */
12  public interface EdgeParser {
13  
14  	/**
15  	 * Whether there's a next edge that can be parsed.
16  	 * @return true if there's a next edge, false otherwise.
17  	 * @throws IOException when an error occurs while reading.
18  	 * @throws InvalidEdgeFormatException when an invalid line is found during reading. This
19  	 * can happen when, for example, the source or destination is not a number, or when the
20  	 * destination is missing.
21  	 */
22  	boolean hasNext() throws IOException, InvalidEdgeFormatException;
23  
24  	/**
25  	 * Returns the next {@link Edge}, or throws a {@link NoSuchElementException}
26  	 * if there's nothing left to be parsed.
27  	 * @return A parsed {@link Edge}.
28  	 * @throws IOException when an error occurs while reading.
29  	 * @throws InvalidEdgeFormatException when an invalid line is found during reading. This
30  	 * can happen when, for example, the source or destination is not a number, or when the
31  	 * destination is missing.
32  	 */
33  	Edge<Integer> next() throws IOException, InvalidEdgeFormatException;
34  
35  }
