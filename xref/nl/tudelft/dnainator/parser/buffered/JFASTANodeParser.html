1   package nl.tudelft.dnainator.parser.buffered;

3   import net.sf.jfasta.FASTAElement;
4   import net.sf.jfasta.FASTAFileReader;
5   import net.sf.jfasta.impl.FASTAElementIterator;
6   import net.sf.jfasta.impl.FASTAFileReaderImpl;
7   import nl.tudelft.dnainator.core.SequenceNode;
8   import nl.tudelft.dnainator.core.SequenceFactory;
9   import nl.tudelft.dnainator.parser.HeaderParser;
10  import nl.tudelft.dnainator.parser.exceptions.InvalidHeaderFormatException;
11  
12  import java.io.BufferedReader;
13  import java.io.IOException;
14  
15  import static java.lang.Integer.parseInt;
16  
17  /**
18   * A {@link BufferedNodeParser} which uses JFASTA's parser internally.
19   */
20  public class JFASTANodeParser extends BufferedNodeParser {
21  
22  	private FASTAFileReader fr;
23  	private FASTAElementIterator it;
24  
25  	/**
26  	 * Constructs a new JFASTAParser.
27  	 *
28  	 * @param br The {@link BufferedReader} from which to read.
29  	 * @param sf The {@link SequenceFactory} used to created {@link SequenceNode}s.
30  	 * @throws IOException If something goes wrong constructing.
31  	 */
32  	public JFASTANodeParser(SequenceFactory sf, BufferedReader br) throws IOException {
33  		super(sf, br);
34  		fr = new FASTAFileReaderImpl(this.br);
35  		it = fr.getIterator();
36  	}
37  
38  	@Override
39  	public boolean hasNext() throws IOException {
40  		return it.hasNext();
41  	}
42  
43  	@Override
44  	public SequenceNode next() throws IOException, InvalidHeaderFormatException {
45  		FASTAElement next = it.next();
46  		HeaderParser p = new HeaderParser(next.getHeader());
47  		SequenceNode s = sf.build(p.next(), p.next(), parseInt(p.next()),
48  				parseInt(p.next()), next.getSequence());
49  		return s;
50  	}
51  
52  }
