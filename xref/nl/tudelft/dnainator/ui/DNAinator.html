1   package nl.tudelft.dnainator.ui;
2   
3   import java.io.IOException;
4   
5   import javafx.application.Application;
6   import javafx.fxml.FXMLLoader;
7   import javafx.scene.Scene;
8   import javafx.scene.image.Image;
9   import javafx.scene.layout.BorderPane;
10  import javafx.stage.Stage;
11  
12  /**
13   * DNAinator's main window, from which all interaction will occur.
14   * <p>
15   * The {@link #start(Stage) start} method will automatically load
16   * the required fxml files, which in turn instantiates all controllers.
17   * </p>
18   */
19  public class DNAinator extends Application {
20  	private static final String DNAINATOR = "DNAinator";
21  	private static final String FXML = "/ui/fxml/dnainator.fxml";
22  	private static final String ICON = "/ui/icons/dnainator.png";
23  
24  	@Override
25  	public void start(Stage primaryStage) throws Exception {
26  		primaryStage.setTitle(DNAINATOR);
27  		primaryStage.getIcons().add(new Image(getClass().getResourceAsStream(ICON)));
28  		primaryStage.setMaximized(true);
29  		try {
30  			BorderPane rootLayout = FXMLLoader.load(getClass().getResource(FXML));
31  			Scene scene = new Scene(rootLayout);
32  			primaryStage.setScene(scene);
33  		} catch (IOException e) {
34  			e.printStackTrace();
35  		}
36  		primaryStage.show();
37  	}
38  }
