3   import javafx.application.Application;
4   import javafx.application.Platform;
5   import javafx.application.Preloader;
6   import javafx.beans.property.BooleanProperty;
7   import javafx.beans.property.SimpleBooleanProperty;
8   import javafx.concurrent.Task;
9   import javafx.fxml.FXMLLoader;
10  import javafx.scene.Scene;
11  import javafx.scene.image.Image;
12  import javafx.scene.layout.BorderPane;
13  import javafx.stage.Screen;
14  import javafx.stage.Stage;
15  import nl.tudelft.dnainator.graph.Graph;
16  import nl.tudelft.dnainator.graph.impl.Neo4jSingleton;
17  import nl.tudelft.dnainator.ui.widgets.dialogs.ExceptionDialog;
18  
19  import java.io.IOException;
20  
21  /**
22   * DNAinator's main window, from which all interaction will occur.
23   * <p>
24   * The {@link #start(Stage) start} method will automatically load
25   * the required fxml files, which in turn instantiates all controllers.
26   * </p>
27   */
28  public class DNAinator extends Application {
29  	private static final String DNAINATOR = "DNAinator";
30  	private static final String FXML = "/ui/fxml/dnainator.fxml";
31  	private static final String ICON = "/ui/dnainator.iconset/icon_512x512.png";
32  
33  	private static final int MIN_WIDTH = 300;
34  	private static final int MIN_HEIGHT = 150;
35  
36  	private BooleanProperty ready = new SimpleBooleanProperty(false);
37  
38  	private void loadDB() {
39  		Task<Graph> task = new Task<Graph>() {
40  			@Override
41  			protected Graph call() throws Exception {
42  				Graph db = Neo4jSingleton.getInstance().getDatabase();
43  				ready.set(true);
44  				return db;
45  			}
46  		};
47  
48  		task.setOnFailed(event -> {
49  			Exception e = new RuntimeException(
50  						"Could not launch the application. Please make sure only\n"
51  								+ "one instance of the application exists at all times.");
52  			notifyPreloader(new Preloader.ErrorNotification("DNAinator",
53  					"Could not launch the application!", e));
54  		});
55  		new Thread(task).start();
56  	}
57  
58  	@Override
59  	public void init() {
60  		loadDB();
61  	}
62  
63  	@Override
64  	public void start(Stage primaryStage) throws Exception {
65  		ready.addListener(event -> {
66  			Platform.runLater(() -> {
67  				primaryStage.setTitle(DNAINATOR);
68  				primaryStage.getIcons().add(new Image(getClass().getResourceAsStream(ICON)));
69  				primaryStage.setMinWidth(MIN_WIDTH);
70  				primaryStage.setMinHeight(MIN_HEIGHT);
71  
72  				notifyPreloader(new Preloader.ProgressNotification(1));
73  
74  				try {
75  					BorderPane rootLayout = FXMLLoader.load(getClass().getResource(FXML));
76  					Scene scene = new Scene(rootLayout, getScreenWidth(), getScreenHeight());
77  					primaryStage.setScene(scene);
78  					primaryStage.show();
79  				} catch (IOException e) {
80  					new ExceptionDialog(null, e, "Could not launch the Application!");
81  				}
82  			});
83  		});
84  	}
85  
86  	private double getScreenWidth() {
87  		return Screen.getPrimary().getVisualBounds().getWidth();
88  	}
89  
90  	private double getScreenHeight() {
91  		return Screen.getPrimary().getVisualBounds().getHeight();
92  	}
93  
94  }
