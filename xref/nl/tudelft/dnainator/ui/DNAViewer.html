3   import nl.tudelft.dnainator.graph.DNAGraph;

6   import org.graphstream.graph.Graph;
7   import org.graphstream.ui.layout.Layout;
8   import org.graphstream.ui.view.Viewer;
9   
10  /**
11   * The viewer is responsible for managing the interactable views of the the strain graph.
12   * Use the addDefaultView() method to obtain a view.
13   */
14  public class DNAViewer extends Viewer {
15  	/**
16  	 * Creates a new GraphViewer.
17  	 */
18  	public DNAViewer() {
19  		this(new DNAGraph(), ThreadingModel.GRAPH_IN_ANOTHER_THREAD);
20  	}
21  
22  	/**
23  	 * Creates a new GraphViewer.
24  	 * @param graph	The graph of strains.
25  	 * @param model	The ThreadingModel to use.
26  	 */
27  	public DNAViewer(Graph graph, ThreadingModel model) {
28  		super(graph, model);
29  		
30  		Layout layout = new DNALayout(graph);
31  		this.enableAutoLayout(layout);
32  		this.setCloseFramePolicy(Viewer.CloseFramePolicy.EXIT);
33  	}
34  }
