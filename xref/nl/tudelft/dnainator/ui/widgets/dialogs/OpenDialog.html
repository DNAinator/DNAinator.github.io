1   package nl.tudelft.dnainator.ui.widgets.dialogs;
2   
3   import javafx.geometry.Insets;
4   import javafx.scene.Node;
5   import javafx.scene.control.Alert;
6   import javafx.scene.control.Button;
7   import javafx.scene.control.ButtonBar;
8   import javafx.scene.control.ButtonType;
9   import javafx.scene.control.Label;
10  import javafx.scene.control.TextField;
11  import javafx.scene.layout.GridPane;
12  import javafx.stage.FileChooser;
13  import nl.tudelft.dnainator.ui.services.GraphLoadService;
14  import nl.tudelft.dnainator.ui.services.NewickLoadService;
15  
16  import java.io.File;
17  
18  /**
19   * Provides a way for the user to open node, edge and newick files in one place.
20   */
21  public class OpenDialog extends Alert {
22  	private static final String EDGE = ".edge.graph";
23  	private static final String NODE = ".node.graph";
24  	private static final String NEWICK = ".nwk";
25  	private static final int TEXTFIELD_WIDTH = 400;
26  	private static final int GRID_PADDING = 10;
27  
28  	private Node root;
29  	private GraphLoadService graphLoadService;
30  	private NewickLoadService newickLoadService;
31  	private TextField nodeField;
32  	private TextField edgeField;
33  	private TextField newickField;
34  	private Button openButton;
35  
36  	/**
37  	 * Creates an OpenDialog of type {@link javafx.scene.control.Alert.AlertType}.NONE.
38  	 * It creates the content of the {@link Alert} and sets up all control.
39  	 *
40  	 * @param root          The root {@link Node} from where this dialog is created.
41  	 * @param graphService  The {@link GraphLoadService} used to generate the graph.
42  	 * @param newickService The {@link NewickLoadService} used to generate the phylogenetic tree.
43  	 */
44  	public OpenDialog(Node root, GraphLoadService graphService,
45  	                  NewickLoadService newickService) {
46  		super(AlertType.NONE);
47  		this.root = root;
48  		this.graphLoadService = graphService;
49  		this.newickLoadService = newickService;
50  
51  		initOwner(root.getScene().getWindow());
52  		setTitle("Open files");
53  		setHeaderText("Please select the files to open.");
54  
55  		ButtonType openButtonType = new ButtonType("Open", ButtonBar.ButtonData.OK_DONE);
56  		getDialogPane().getButtonTypes().addAll(ButtonType.CANCEL, openButtonType);
57  		openButton = (Button) getDialogPane().lookupButton(openButtonType);
58  		openButton.setDisable(true);
59  
60  		setupGrid();
61  	}
62  
63  	private void setupGrid() {
64  		GridPane grid = new GridPane();
65  		grid.setVgap(GRID_PADDING);
66  		grid.setHgap(GRID_PADDING);
67  		grid.setPadding(new Insets(GRID_PADDING));
68  
69  		setupTextFields(grid);
70  		setupLabels(grid);
71  
72  		getDialogPane().setContent(grid);
73  	}
74  
75  	private void setupTextFields(GridPane grid) {
76  		nodeField = new TextField();
77  		nodeField.setPrefWidth(TEXTFIELD_WIDTH);
78  		nodeField.setPromptText("Node file (*" + NODE + ")");
79  		nodeField.setOnMouseClicked(e -> browseNodeListener());
80  
81  		edgeField = new TextField();
82  		edgeField.setPromptText("Edge file (*" + EDGE + ")");
83  		edgeField.setOnMouseClicked(e -> browseEdgeListener());
84  
85  		newickField = new TextField();
86  		newickField.setPromptText("Tree file (*" + NEWICK + ")");
87  		newickField.setOnMouseClicked(e -> browseNewickListener());
88  
89  		grid.add(nodeField, 1, 0);
90  		grid.add(edgeField, 1, 1);
91  		grid.add(newickField, 1, 2);
92  	}
93  
94  	private void setupLabels(GridPane grid) {
95  		grid.add(new Label("Nodes:"), 0, 0);
96  		grid.add(new Label("Edges:"), 0, 1);
97  		grid.add(new Label("Tree:"), 0, 2);
98  	}
99  
100 	private void browseNodeListener() {
101 		File nodeFile = genFileChooser("Open node file", NODE)
102 				.showOpenDialog(root.getScene().getWindow());
103 		updateForNode(nodeFile);
104 	}
105 
106 	private void browseEdgeListener() {
107 		File edgeFile = genFileChooser("Open edge file", EDGE)
108 				.showOpenDialog(root.getScene().getWindow());
109 		updateForEdge(edgeFile);
110 	}
111 
112 	private void browseNewickListener() {
113 		File newickFile = genFileChooser("Open newick file", NEWICK)
114 				.showOpenDialog(root.getScene().getWindow());
115 		updateForNewick(newickFile);
116 	}
117 
118 	private void updateForNode(File nodeFile) {
119 		if (nodeFile != null) {
120 			graphLoadService.setNodeFile(nodeFile);
121 			nodeField.setText(graphLoadService.getNodeFile().getAbsolutePath());
122 			graphLoadService.setEdgeFile(openEdgeFile(nodeFile.getPath()));
123 			edgeField.setText(graphLoadService.getEdgeFile().getAbsolutePath());
124 		}
125 
126 		updateOpenButton();
127 	}
128 
129 	private void updateForEdge(File edgeFile) {
130 		if (edgeFile != null) {
131 			graphLoadService.setEdgeFile(edgeFile);
132 			edgeField.setText(graphLoadService.getEdgeFile().getAbsolutePath());
133 			graphLoadService.setNodeFile(openNodeFile(edgeFile.getPath()));
134 			nodeField.setText(graphLoadService.getNodeFile().getAbsolutePath());
135 		}
136 
137 		updateOpenButton();
138 	}
139 
140 	private void updateForNewick(File newickFile) {
141 		if (newickFile != null) {
142 			newickLoadService.setNewickFile(newickFile);
143 			newickField.setText(newickLoadService.getNewickFile().getAbsolutePath());
144 		}
145 
146 		updateOpenButton();
147 	}
148 
149 	private FileChooser genFileChooser(String title, String extension) {
150 		FileChooser fileChooser = new FileChooser();
151 		fileChooser.setTitle(title);
152 		fileChooser.getExtensionFilters().add(
153 				new FileChooser.ExtensionFilter(title, "*" + extension));
154 		return fileChooser;
155 	}
156 
157 	private void updateOpenButton() {
158 		boolean isNodeFileSelected = graphLoadService.getNodeFile() != null;
159 		boolean isEdgeFileSelected = graphLoadService.getEdgeFile() != null;
160 		boolean isNewickFileSelected = newickLoadService.getNewickFile() != null;
161 
162 		if (isNewickFileSelected || (isNodeFileSelected && isEdgeFileSelected)) {
163 			openButton.setDisable(false);
164 		} else {
165 			openButton.setDisable(true);
166 		}
167 	}
168 
169 	private File openEdgeFile(String path) {
170 		return new File(path.substring(0, path.length() - NODE.length()).concat(EDGE));
171 	}
172 
173 	private File openNodeFile(String path) {
174 		return new File(path.substring(0, path.length() - EDGE.length()).concat(NODE));
175 	}
176 }
