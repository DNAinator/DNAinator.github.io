8   import javafx.geometry.Bounds;
9   import javafx.scene.paint.Color;
10  import javafx.scene.shape.Rectangle;
11  import nl.tudelft.dnainator.core.impl.Cluster;
12  import nl.tudelft.dnainator.graph.Graph;
13  import nl.tudelft.dnainator.graph.impl.Neo4jSingleton;
14  
15  /**
16   * The {@link GraphItem} class represents the top level object in the viewable model.
17   * It is a {@link CompositeItem}, that can hold both content and children.
18   */
19  public class GraphItem extends CompositeItem {
20  	private static final int FOUR = 4;
21  	private static final int CLUSTER_DIVIDER = 100;
22  
23  	private Graph graph;
24  	private Map<String, ClusterItem> clusters;
25  
26  	/**
27  	 * Construct a new top level {@link GraphItem} using the default graph.
28  	 */
29  	public GraphItem() {
30  		this(Neo4jSingleton.getInstance().getDatabase());
31  	}
32  
33  	/**
34  	 * Construct a new top level {@link GraphItem} using the specified graph.
35  	 * @param graph	the specified graph
36  	 */
37  	public GraphItem(Graph graph) {
38  		super(null, 0);
39  		this.graph = graph;
40  		this.clusters = new HashMap<>();
41  
42  		for (int i = 0; i < FOUR; i++) {
43  			int width = NO_CLUSTERS * RANK_WIDTH / FOUR;
44  			Rectangle r = new Rectangle(width, RANK_WIDTH, Color.BLACK);
45  			r.setTranslateX(i * width);
46  			getContent().getChildren().add(r);
47  		}
48  	}
49  
50  	@Override
51  	public Graph getGraph() {
52  		return graph;
53  	}
54  
55  	@Override
56  	public Map<String, ClusterItem> getClusters() {
57  		return clusters;
58  	}
59  
60  	@Override
61  	public ModelItem getRoot() {
62  		return this;
63  	}
64  
65  	@Override
66  	public void loadChildren(Bounds b) {
67  		System.out.println("load iteration");
68  		getChildItems().clear();
69  		getChildContent().getChildren().clear();
70  
71  		int minrank = (int) (Math.max(b.getMinX() / RANK_WIDTH, 0));
72  		int maxrank = (int) (b.getMaxX() / RANK_WIDTH);
73  		List<String> roots = getGraph().getRank(minrank).stream()
74  						.map(e -> e.getId()).collect(Collectors.toList());
75  		System.out.println(b.getWidth());
76  		Map<Integer, List<Cluster>> clusters = getGraph().getAllClusters(roots, maxrank,
77  										(int) (b.getWidth() / CLUSTER_DIVIDER));
78  
79  		clusters.forEach((k, v) -> getChildItems().add(new RankItem(this, k, v)));
80  	};
81  
82  	@Override
83  	public void update(Bounds b) {
84  		update(b, Thresholds.GRAPH);
85  	}
86  }
