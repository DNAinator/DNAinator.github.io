3   import javafx.geometry.Bounds;
4   import javafx.scene.paint.Color;
5   import javafx.scene.shape.Circle;
6   
7   /**
8    * The {@link ClusterItem} class represents the mid level object in the viewable model.
9    * Just like the {@link GraphItem} class, it's a {@link CompositeItem},
10   * that can hold both content and children.
11   */
12  public class ClusterItem extends CompositeItem {
13  	/**
14  	 * Construct a new mid level {@link ClusterItem} using the default graph.
15  	 * Every {@link ClusterItem} needs a reference to its parent.
16  	 * @param parent	the parent of this {@link ClusterItem}
17  	 */
18  	public ClusterItem(ModelItem parent) {
19  		super(parent);
20  
21  		bindLocalToRoot(parent.localToRootProperty());
22  
23  		getContent().getChildren().add(new Circle(CLUSTER_SIZE, Color.BLUE));
24  
25  		// FIXME: These should be lazily instantiated!
26  		for (int i = 0; i < NO_RANKS; i++) {
27  			RankItem si = new RankItem(this);
28  			si.setTranslateX(i * RANK_WIDTH);
29  			getChildItems().add(si);
30  		}
31  	}
32  
33  	@Override
34  	public void update(Bounds b) {
35  		if (!isInViewport(b)) {
36  			return;
37  		}
38  
39  		update(b, Thresholds.CLUSTER);
40  	}
41  
42  }
