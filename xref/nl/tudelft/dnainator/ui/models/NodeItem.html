1   package nl.tudelft.dnainator.ui.models;
2   
3   import javafx.geometry.Bounds;
4   import javafx.scene.Group;
5   import javafx.scene.paint.Color;
6   import javafx.scene.paint.Paint;
7   import javafx.scene.shape.Circle;
8   import nl.tudelft.dnainator.core.SequenceNode;
9   import nl.tudelft.dnainator.ui.drawables.DrawableEdge;
10  import nl.tudelft.dnainator.ui.widgets.contexts.NodeContext;
11  
12  /**
13   * The drawable node is the JavaFX counterpart of {@link SequenceNode}.
14   */
15  public class NodeItem extends ModelItem {
16  	private static final double RADIUS = 3;
17  	private static final Paint FILL = Color.rgb(242, 173, 12);
18  	private SequenceNode node;
19  	// Should be moved to ModelItem
20  	private Group edges;
21  
22  	/**
23  	 * Instantiate a new NodeItem with default radius and default fill.
24  	 * @param parent	the parent of this {@link NodeItem}
25  	 * @param node		The {@link SequenceNode} this NodeItem represents.
26  	 */
27  	public NodeItem(ModelItem parent, SequenceNode node) {
28  		this(parent, node, RADIUS, FILL);
29  	}
30  
31  	/**
32  	 * Instantiate a new NodeItem with default fill.
33  	 * @param parent	the parent of this {@link NodeItem}
34  	 * @param node		The {@link SequenceNode} this NodeItem represents.
35  	 * @param radius	This NodeItem's radius.
36  	 */
37  	public NodeItem(ModelItem parent, SequenceNode node, double radius) {
38  		this(parent, node, radius, FILL);
39  	}
40  
41  	/**
42  	 * Instantiate a new NodeItem.
43  	 * @param parent	the parent of this {@link NodeItem}
44  	 * @param node		The {@link SequenceNode} this NodeItem represents.
45  	 * @param radius	This NodeItem's radius.
46  	 * @param fill		This NodeItem's interior color.
47  	 */
48  	public NodeItem(ModelItem parent, SequenceNode node, double radius, Paint fill) {
49  		super(parent);
50  		this.node = node;
51  		this.edges = new Group();
52  
53  		bindLocalToRoot(parent.localToRootProperty());
54  
55  		getStyleClass().add("drawable-node");
56  		getContent().getChildren().add(edges);
57  		getContent().getChildren().add(new Circle(radius, fill));
58  		setOnContextMenuRequested(e -> {
59  			NodeContext.getInstance().show(NodeItem.this, e.getScreenX(), e.getScreenY());
60  			e.consume();
61  		});
62  	}
63  
64  	/**
65  	 * @return This NodeItem's {@link SequenceNode}.
66  	 */
67  	public SequenceNode getSequenceNode() {
68  		return node;
69  	}
70  
71  	@Override
72  	public void update(Bounds b) {
73  		if (edges.getChildren().size() < getSequenceNode().getIncoming().size()) {
74  			for (String e : node.getIncoming()) {
75  				NodeItem o = getNodes().get(e);
76  				if (o != null) {
77  					edges.getChildren().add(new DrawableEdge(this, o));
78  				}
79  			}
80  		}
81  	}
82  }
