1   package nl.tudelft.dnainator.ui.splashscreen;
2   
3   import java.io.IOException;
4   import java.io.InputStream;
5   import java.util.Properties;
6   
7   import nl.tudelft.dnainator.ui.widgets.dialogs.ExceptionDialog;
8   import javafx.application.Preloader;
9   import javafx.scene.Scene;
10  import javafx.scene.control.ProgressBar;
11  import javafx.scene.image.Image;
12  import javafx.scene.image.ImageView;
13  import javafx.scene.layout.VBox;
14  import javafx.scene.text.Text;
15  import javafx.stage.Stage;
16  import javafx.stage.StageStyle;
17  
18  /**
19   * A splashscreen to display while the application is initialising.
20   */
21  public class SplashScreen extends Preloader {
22  	private static final String SPLASH_IMG = "/ui/splashscreen/splashscreen.png";
23  	private static final String PROPERTIES = "config.properties";
24  	private static final String STYLE = "/ui/style.css";
25  	private static final int SPLASH_WIDTH = 620;
26  	private static final int SPLASH_HEIGHT = 317;
27  
28  	private Stage stage;
29  	private ProgressBar progressBar;
30  	private Properties prop;
31  
32  	@Override
33  	public void start(Stage primaryStage) throws Exception {
34  		stage = primaryStage;
35  		stage.initStyle(StageStyle.UNDECORATED);
36  		stage.setScene(createPreloaderScene());
37  		stage.show();
38  	}
39  
40  	@Override
41  	public void handleApplicationNotification(PreloaderNotification info) {
42  		stage.hide();
43  	}
44  
45  	private Scene createPreloaderScene() {
46  		VBox container = new VBox();
47  		container.getStylesheets().add(getClass().getResource(STYLE).toString());
48  		ImageView imageView = new ImageView(new Image(SPLASH_IMG));
49  
50  		progressBar = new ProgressBar();
51  		progressBar.setPrefWidth(SPLASH_WIDTH);
52  		
53  		Text version = new Text();
54  		readProperties();
55  		version.setText("Version: " + prop.getProperty("version"));
56  		
57  		// CHECKSTYLE.OFF: MagicNumber
58  		version.setTranslateX(167);
59  		version.setTranslateY(-150);
60  		// CHECKSTYLE.ON: MagicNumber
61  		
62  		version.getStyleClass().add("splash-version");
63  		container.getChildren().addAll(imageView, progressBar, version);
64  		
65  		return new Scene(container, SPLASH_WIDTH, SPLASH_HEIGHT);
66  	}
67  	
68  	private void readProperties() {
69  		prop = new Properties();
70  		try (InputStream in = getClass().getClassLoader().getResourceAsStream(PROPERTIES)) {
71  			prop.load(in);
72  		} catch (IOException e) {
73  			new ExceptionDialog(null, e, "Error reading properties.");
74  		}
75  	}
76  }
