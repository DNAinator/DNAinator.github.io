1   package nl.tudelft.dnainator.ui.drawables.phylogeny;
2   
3   import javafx.scene.shape.HLineTo;
4   import javafx.scene.shape.MoveTo;
5   import javafx.scene.shape.Path;
6   import javafx.scene.shape.VLineTo;
7   
8   /**
9    * An edge for use in a phylogenetic tree.
10   */
11  public class Edge extends Path {
12  	private HLineTo hline;
13  	private VLineTo vline;
14  
15  	/**
16  	 * Constructs a new {@link Edge}, ending at the provided {@link AbstractNode}.
17  	 * @param dst This edge's destination node.
18  	 */
19  	public Edge(AbstractNode dst) {
20  		// Set start point of the Path.
21  		MoveTo m = new MoveTo();
22  		m.xProperty().bindBidirectional(dst.centerXProperty());
23  		m.yProperty().bindBidirectional(dst.centerYProperty());
24  		getElements().add(m);
25  
26  		// Add the horizontal line, starting at the path's start point.
27  		hline = new HLineTo();
28  		getElements().add(hline);
29  
30  		// Add the vertical line, starting at the horizontal line's end point.
31  		vline = new VLineTo();
32  		getElements().add(vline);
33  
34  		getStyleClass().add("phylogenetic-edge");
35  	}
36  
37  	/**
38  	 * Binds this {@link Edge}'s vertical and horizontal lines' end points to <code>src</code>.
39  	 * @param src The {@link AbstractNode} to bind the end points to.
40  	 */
41  	public void bindTo(AbstractNode src) {
42  		vline.yProperty().bindBidirectional(src.centerYProperty());
43  		hline.xProperty().bindBidirectional(src.centerXProperty());
44  	}
45  }
