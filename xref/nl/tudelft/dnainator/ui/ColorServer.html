1   package nl.tudelft.dnainator.ui;
2   
3   import java.util.BitSet;
4   import java.util.HashMap;
5   import java.util.Map;
6   
7   /**
8    * The {@link ColorServer} is responsible for managing colors used in the phylogenetic tree and the
9    * DNA strains.
10   *
11   * It keeps a {@link HashMap} of strain sources to CSS classes and uses a {@link BitSet} as a list
12   * of occupied colors.
13   */
14  public final class ColorServer {
15  	private static final String COLOR = "color-";
16  	private static final int COLORS = 21;
17  	private static ColorServer instance = new ColorServer();
18  	private	Map<String, String> colors;
19  	private BitSet occupied;
20  
21  	private ColorServer() {
22  		colors = new HashMap<>(COLORS);
23  		occupied = new BitSet(COLORS);
24  	}
25  
26  	/**
27  	 * @param source The source of the strain to highlight.
28  	 * @return The CSS class for an unused color.
29  	 * @throws AllColorsInUseException If all colors are currently in use.
30  	 */
31  	public String getColor(String source) throws AllColorsInUseException {
32  		if (colors.containsKey(source)) {
33  			return colors.get(source);
34  		} else if (occupied.cardinality() == COLORS) {
35  			throw new AllColorsInUseException();
36  		}
37  
38  		int index = occupied.nextClearBit(0);
39  		occupied.set(index, true);
40  		String res = COLOR + index;
41  		colors.put(source, res);
42  		return res;
43  	}
44  
45  	/**
46  	 * Tells the {@link ColorServer} that the color belonging to the source is not used anymore.
47  	 * @param source The source whose color is now unused.
48  	 */
49  	public void revokeColor(String source) {
50  		if (!colors.containsKey(source)) {
51  			return;
52  		}
53  
54  		String color = colors.remove(source);
55  		int index = Character.getNumericValue(color.charAt(color.length() - 1));
56  		occupied.set(index, false);
57  	}
58  
59  	/**
60  	 * @return The {@link ColorServer} instance.
61  	 */
62  	public static ColorServer getInstance() {
63  		return instance;
64  	}
65  }
