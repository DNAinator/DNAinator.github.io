24  /**
25   * This class is the View part of the MVC pattern.
26   */
27  public class View extends Pane {
28  	@FXML
29  	private BorderPane root;
30  
31  	private Scale scale;
32  	private Translate toCenter;
33  	private Translate translate;
34  	private Transform worldToCamera;
35  
36  	private ModelItem mi;
37  
38  	/**
39  	 * Creates a new view instance.
40  	 */
41  	public View() {
42  		try {
43  			FileUtils.deleteRecursively(new File(Neo4jSingleton.DB_PATH));
44  		} catch (IOException e) {
45  			e.printStackTrace();
46  		}
47  
48  		loadFXML();
49  		getStyleClass().add("view");
50  		setOnContextMenuRequested(e -> {
51  			ViewContext.getInstance().show(View.this, e.getScreenX(), e.getScreenY());
52  			e.consume();
53  		});
54  
55  		toCenter = new Translate();
56  		widthProperty().addListener((o, v1, v2) -> toCenter.setX(v2.intValue() / 2));
57  		heightProperty().addListener((o, v1, v2) -> toCenter.setY(v2.intValue() / 2));
58  
59  		translate = new Translate();
60  		translate.setOnTransformChanged(e -> worldToCamera = worldToCamera());
61  
62  		scale = new Scale();
63  		scale.setOnTransformChanged(e -> worldToCamera = worldToCamera());
64  
65  		mi = new GraphItem();
66  		mi.getTransforms().add(toCenter);
67  		mi.getTransforms().add(scale);
68  		mi.getTransforms().add(translate);
69  		getChildren().add(mi);
70  	}
71  
72  	private void loadFXML() {
73  		FXMLLoader fxmlLoader = new FXMLLoader(getClass().getResource("/ui/fxml/view.fxml"));
74  		fxmlLoader.setRoot(this);
75  
76  		try {
77  			fxmlLoader.load();
78  		} catch (IOException e) {
79  			new ExceptionDialog(root, e, "Can not load the View!");
80  		}
81  	}
82  
83  	/**
84  	 * Returns the concatenated transform from world coordinates to camera coordinates.
85  	 * @return	the concatenated transform
86  	 */
87  	private Transform worldToCamera() {
88  		return toCenter.createConcatenation(scale).createConcatenation(translate);
89  	}
90  
91  	/**
92  	 * Transforms a given bounding box from camera coordinates to world coordinates.
93  	 * @param b	the given bounding box
94  	 * @return	the transformed bounding box
95  	 */
96  	public Bounds cameraToWorld(Bounds b) {
97  		Bounds world = null;
98  		try {
99  			world = worldToCamera.inverseTransform(b);
100 		} catch (NonInvertibleTransformException e) {
101 			e.printStackTrace();
102 		}
103 		return world;
104 	}
105 
106 	/**
107 	 * Pan the camera by the amount given by the delta vector.
108 	 * FIXME: sensitivity depends on zoom level.
109 	 * @param delta	the delta vector
110 	 */
111 	public void pan(Point2D delta) {
112 		try {
113 			translate.setX(translate.getX() + scale.inverseTransform(delta).getX());
114 			translate.setY(translate.getY() + scale.inverseTransform(delta).getY());
115 		} catch (NonInvertibleTransformException e) {
116 			e.printStackTrace();
117 		}
118 		mi.update(cameraToWorld(getLayoutBounds()));
119 	}
120 
121 	/**
122 	 * Zoom the camera by the amount given by zoom.
123 	 * @param zoom	the amount to zoom in
124 	 */
125 	public void zoom(Double zoom) {
126 		scale.setX(scale.getX() + (scale.getX() * zoom));
127 		scale.setY(scale.getY() + (scale.getY() * zoom));
128 		mi.update(cameraToWorld(getLayoutBounds()));
129 	}
130 }
