23  /**
24   * This class is instantiated by a {@link GraphLoadService} in order to load a sequence graph.
25   */
26  public class GraphLoadTask extends Task<Graph> {
27  	private static final int STEPS = 5;
28  	private GraphLoadService service;
29  
30  	/**
31  	 * Create a new GraphLoadTask, which will be run by the specified service.
32  	 * @param service	the service
33  	 */
34  	public GraphLoadTask(GraphLoadService service) {
35  		this.service = service;
36  	}
37  
38  	/**
39  	 * Construct an phylogeny tree by filling it with the newick files of the executing service.
40  	 * @return		the phylogeny tree
41  	 * @throws IOException	when the newick file was invalid
42  	 */
43  	private TreeNode constructTree() throws IOException {
44  		return new TreeParser(service.getNewickFile()).parse();
45  	}
46  
47  	/**
48  	 * Construct an annotation collection by filling it with
49  	 * the annotation and DR mutation files of the executing service.
50  	 * @return		the filled annotation collection
51  	 * @throws IOException	when the annotation or mutation file was invalid
52  	 */
53  	private AnnotationCollection constructAnnotations() throws IOException {
54  		AnnotationCollection annotations;
55  		annotations = new AnnotationCollectionImpl(new AnnotationIterator(service.getGffFile()));
56  		if (service.getDrFile() != null) {
57  			annotations.addAnnotations(new DRMutationIterator(service.getDrFile()));
58  		}
59  		return annotations;
60  	}
61  
62  	/**
63  	 * Construct a graph by filling the specified graph builder
64  	 * with the node and edge files of the executing service.
65  	 * @param builder	the graph builder
66  	 * @return		the constructed graph builder
67  	 * @throws IOException	when the edge or node file was invalid
68  	 */
69  	private GraphBuilder constructGraph(GraphBuilder builder) throws IOException {
70  		Iterator<SequenceNode> np = new NodeIterator(service.getNodeFile());
71  		Iterator<Edge<?>> ep = new EdgeIterator(service.getEdgeFile());
72  		return builder.constructGraph(np, ep);
73  	}
74  	
75  	@Override
76  	protected Graph call() throws IOException, ParseException {
77  		if (!service.canLoad()) {
78  			throw new IOException("Not all required fields are filled in!");
79  		}
80  		int progress = 0;
81  
82  		TreeNode node = constructTree();
83  		updateProgress(++progress, STEPS);
84  
85  		AnnotationCollection annotations = constructAnnotations();
86  		updateProgress(++progress, STEPS);
87  
88  		// This constructs a graph builder, which is used to insert all nodes in one batch.
89  		// This has significantly improved loading times.
90  		GraphBuilder builder = new Neo4jBatchBuilder(service.getDatabase(), annotations, node);
91  		updateProgress(++progress, STEPS);
92  
93  		constructGraph(builder);
94  		updateProgress(++progress, STEPS);
95  
96  		Graph graph = builder.build();
97  		updateProgress(++progress, STEPS);
98  
99  		return graph;
100 	}
101 }
