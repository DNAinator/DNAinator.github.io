1   package nl.tudelft.dnainator.javafx.widgets.dialogs;
2   
3   import javafx.scene.Node;
4   import javafx.scene.control.Alert;
5   import javafx.scene.control.ButtonBar.ButtonData;
6   import javafx.scene.control.ButtonType;
7   import javafx.scene.image.Image;
8   import javafx.scene.image.ImageView;
9   
10  import java.io.IOException;
11  import java.io.InputStream;
12  import java.util.Properties;
13  
14  /**
15   * Creates a {@link Alert} displaying information about the
16   * application.
17   */
18  public class AboutDialog extends Alert {
19  	private Node parent;
20  	private Properties prop;
21  	private static final String LOGO = "/dnainator.iconset/icon_128x128.png";
22  	private static final String STYLE = "/style.css";
23  	private static final String PROPERTIES = "/config.properties";
24  	
25  	/**
26  	 * Instantiates a new AboutDialog.
27  	 * It will set up all the necessities for displaying 
28  	 * information of the application.
29  	 * @param parent The parent {@link Node} of this dialog.
30  	 */
31  	public AboutDialog(Node parent) {
32  		super(Alert.AlertType.NONE);
33  		this.parent = parent;
34  		setupDialog();
35  	}
36  
37  	private void setupDialog() {
38  		this.setTitle("About DNAinator");
39  		this.setResizable(true);
40  		this.getDialogPane().getStylesheets().add(getClass().getResource(STYLE).toString());
41  		this.initOwner(parent.getScene().getWindow());
42  		
43  		ImageView img = new ImageView();
44  		img.setImage(new Image(getClass().getResourceAsStream(LOGO)));
45  		this.setGraphic(img);
46  		readProperties();
47  		
48  		this.setHeaderText("DNAinator\nDNA network visualization tool");
49  		this.setContentText(contentText());
50  		
51  		ButtonType close = new ButtonType("Close", ButtonData.CANCEL_CLOSE);
52  		this.getDialogPane().getButtonTypes().add(close);
53  	}
54  
55  	private String contentText() {
56  		StringBuilder contents = new StringBuilder();
57  		contents.append("Version: ")
58  				.append(prop.getProperty("version"))
59  				.append("\n")
60  				.append("Jente Hidskes, Gerlof Fokkema, Owen Huang, "
61  					+ "Skip Lentz, Piet van Agtmaal");
62  		return contents.toString();
63  	}
64  	
65  	private void readProperties() {
66  		prop = new Properties();
67  		try (InputStream in = getClass().getResourceAsStream(PROPERTIES)) {
68  			prop.load(in);
69  		} catch (IOException e) {
70  			new ExceptionDialog(parent, e, "Error reading properties.");
71  		}
72  	}
73  }
