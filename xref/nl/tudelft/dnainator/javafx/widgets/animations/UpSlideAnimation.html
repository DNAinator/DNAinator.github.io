3   import javafx.scene.layout.Pane;

5   /**
6    * The {@link UpSlideAnimation} is a {@link SlidingAnimation} that transitions upwards.
7    */
8   public class UpSlideAnimation extends SlidingAnimation {
9   
10  	/**
11  	 * Construct a new {@link UpSlideAnimation}, which will animate a sliding
12  	 * transition to upwards.
13  	 * @param pane         The {@link Pane} to be animated.
14  	 * @param size         The size over which the animation will occur.
15  	 * @param duration     The duration of the animations.
16  	 * @param pos          The position of the {@link Pane}.
17  	 */
18  	public UpSlideAnimation(Pane pane, double size, double duration, Position pos) {
19  		super(pane, size, duration, pos);
20  	}
21  
22  	@Override
23  	public void setCurSize(double frac) {
24  		if (pos == Position.TOP) {
25  			newSize = size * (1.0 - frac);
26  		} else if (pos == Position.BOTTOM) {
27  			newSize = size * frac;
28  		}
29  	}
30  }
