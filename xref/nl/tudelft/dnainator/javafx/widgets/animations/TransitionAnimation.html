1   package nl.tudelft.dnainator.javafx.widgets.animations;
2   
3   import javafx.animation.Transition;
4   import javafx.scene.layout.Pane;
5   
6   /**
7    * A {@link CustomAnimation} which animates a transition.
8    * It applies the animation onto a {@link Pane}.
9    */
10  public abstract class TransitionAnimation extends Transition implements CustomAnimation {
11  	
12  	/**
13  	 * Enum to provide the position of the {@link Pane} that will be animated.
14  	 */
15  	public enum Position {
16  		TOP,
17  		BOTTOM,
18  		LEFT,
19  		RIGHT
20  	}
21  	
22  	protected Pane pane;
23  	protected double size;
24  	protected double duration;
25  	protected Position pos;
26  
27  	/**
28  	 * Construct a new {@link TransitionAnimation}.
29  	 * @param pane         The {@link Pane} which is animated.
30  	 * @param size         The length over which the pane will slide.
31  	 * @param duration     The duration of the animations.
32  	 * @param pos          The position of the {@link Pane}.
33  	 */
34  	public TransitionAnimation(Pane pane, double size, double duration, Position pos) {
35  		this.pane = pane;
36  		this.size = size;
37  		this.duration = duration;
38  		this.pos = pos;
39  		this.setOnFinished(actionEvent -> {
40  			pane.setVisible(true);
41  			setRate(-getRate());
42  		});
43  		setupAnimation();
44  	}
45  	
46  	/**
47  	 * Plays the {@link TransitionAnimation}.
48  	 */
49  	@Override
50  	public void playAnimation() {
51  		this.play();
52  	}
53  	
54  	/**
55  	 * Stops the {@link TransitionAnimation}.
56  	 */
57  	@Override
58  	public void stopAnimation() {
59  		this.stop();
60  	}
61  }
