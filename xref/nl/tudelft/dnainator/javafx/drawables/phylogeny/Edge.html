3   import javafx.scene.shape.HLineTo;
4   import javafx.scene.shape.MoveTo;
5   import javafx.scene.shape.Path;
6   import javafx.scene.shape.VLineTo;
7   
8   /**
9    * An edge for use in a phylogenetic tree.
10   */
11  public class Edge extends Path {
12  	/**
13  	 * Constructs a new {@link Edge}, ending at the provided {@link AbstractNode}.
14  	 * @param dst This edge's destination node.
15  	 */
16  	public Edge(AbstractNode dst) {
17  		// Set start point of the Path.
18  		MoveTo m = new MoveTo();
19  		m.xProperty().bindBidirectional(dst.translateXProperty());
20  		m.yProperty().bindBidirectional(dst.translateYProperty());
21  		getElements().add(m);
22  
23  		// Add the horizontal line, starting at the path's start point.
24  		HLineTo hline = new HLineTo();
25  		getElements().add(hline);
26  
27  		// Add the vertical line, starting at the horizontal line's end point.
28  		VLineTo vline = new VLineTo();
29  		getElements().add(vline);
30  	}
31  }
