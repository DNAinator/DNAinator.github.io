1   package nl.tudelft.dnainator.core.graph;
2   
3   import nl.tudelft.dnainator.core.SequenceNode;
4   import nl.tudelft.dnainator.util.Edge;
5   
6   import org.graphstream.stream.Sink;
7   import org.graphstream.stream.SourceBase;
8   
9   /**
10   * This class realizes a graphfactory using GraphStream as it's backend.
11   */
12  public class GSGraphFactory extends SourceBase implements GraphBuilder {
13  	private int edgecount;
14  	
15  	/**
16  	 * This factory constructs a GraphStream graph with the given title.
17  	 * @param g		the initial sink of the graphfactory
18  	 */
19  	public GSGraphFactory(Sink g) {
20  		this.addSink(g);
21  	}
22  	
23  	@Override
24  	public void addEdge(Edge<Integer> e) {
25  		sendEdgeAdded(this.sourceId, Integer.toString(edgecount++),
26  					  Integer.toString(e.getSource()), Integer.toString(e.getDest()), false);
27  	}
28  
29  	@Override
30  	public void addNode(SequenceNode s) {
31  		String id = Integer.toString(s.getId());
32  		sendNodeAdded(this.sourceId, id);
33  		sendNodeAttributeAdded(this.sourceId, id, "start", s.getStartRef());
34  		sendNodeAttributeAdded(this.sourceId, id, "end", s.getEndRef());
35  		sendNodeAttributeAdded(this.sourceId, id, "sequence", s.getSource());
36  		sendNodeAttributeAdded(this.sourceId, id, "source", s.getSequence());
37  	}
38  }
