20  public class ExceptionDialogTest extends ApplicationTest {
21  	
22  	private ExceptionDialog ed;
23  	private Pane pane;
24  	@Mock private Throwable throwable;
25

28  	private Scene scene;
29  	
30  	@Override

34  	}
35  	
36  	/**
37  	 * Set up common variables.
38  	 */
39  	@Before
40  	public void setup() {

43  	}
44  	
45  	/**
46  	 * Test creating a {@link AboutDialog}.
47  	 * Ensure that there is a button and that text is set correctly.
48  	 */
49  	@Test
50  	public void testCreateMenus() {
51  		Platform.runLater(() -> {
52  			ed = new ExceptionDialog(pane, throwable, "msg");
53  			// CHECKSTYLE.OFF: MagicNumber
54  			assertEquals(1, ed.getButtonTypes().size());
55  			// CHECKSTYLE.ON: MagicNumber
56  			assertEquals("msg", ed.getTitle());
57  			assertEquals("error!", ed.getHeaderText());
58  			assertEquals("The exception stacktrace was:", ed.getContentText());
59  		});
60  	}
