3   import javafx.concurrent.Service;

5   import java.util.concurrent.CompletableFuture;

7   /**
8    * Utility class providing useful methods for testing loadservices.
9    */
10  public final class LoadServiceTestUtils {
11  
12  	private LoadServiceTestUtils() {
13  	}
14  	/**
15  	 * Registers a listener on the service to update the completableFuture.
16  	 * @param service The service.
17  	 * @param completableFuture	The future.
18  	 * @param <T> The expected type.
19  	 */
20  	public static <T> void registerListeners(Service<T> service,
21  			CompletableFuture<T> completableFuture) {
22  		service.setOnSucceeded(e -> completableFuture.complete(service.getValue()));
23  		service.setOnFailed(e ->
24  			completableFuture.completeExceptionally(service.getException()));
25  		service.setOnCancelled(e -> completableFuture.cancel(true));
26  	}
27  
28  }
