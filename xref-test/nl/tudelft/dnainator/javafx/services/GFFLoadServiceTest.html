29  /**
30   * Test the loadservice for loading GFF3 files.
31   */
32  @RunWith(JfxRunner.class)
33  public class GFFLoadServiceTest {
34  	private static final String DB_PATH = "target/neo4j-junit-gffload";
35  	private Graph graph;
36  	private GFFLoadService loadService;
37  	private String gffFilePath;
38  	private static final int DELAY = 20000;
39  
40  	/**
41  	 * Setup the database and construct the graph.
42  	 */
43  	@BeforeClass
44  	public static void setUp() {
45  		try {
46  			FileUtils.deleteRecursively(new File(DB_PATH));
47  		} catch (IOException e) {
48  			e.printStackTrace();
49  		}
50  	}
51  
52  	/**
53  	 * Creates test node and edge files.
54  	 * @throws URISyntaxException 
55  	 * @throws ParseException 
56  	 * @throws IOException 
57  	 */
58  	@Before
59  	public void setup() throws URISyntaxException, IOException, ParseException {
60  		File nodeFile = new File(getClass().getResource("/strains/test.node.graph").toURI());
61  		File edgeFile = new File(getClass().getResource("/strains/test.edge.graph").toURI());
62  		new Neo4jBatchBuilder(DB_PATH).constructGraph(new NodeParserImpl(nodeFile),
63  				new EdgeParserImpl(edgeFile));
64  		graph = new Neo4jGraph(DB_PATH);
65  		loadService = new GFFLoadService();
66  		gffFilePath = new File(getClass().getResource("/annotations/test.gff").toURI()).toString();
67  		loadService.setGffFilePath(gffFilePath);
68  	}
69  
70  	/**
71  	 * Test loading of a proper file.
72  	 * @throws InterruptedException 
73  	 * @throws ExecutionException 
74  	 * @throws TimeoutException 
75  	 */
76  	@Test
77  	public void testProperFile() throws InterruptedException, ExecutionException, TimeoutException {
78  		doTest();
79  	}
80  
81  	/**
82  	 * Performs the actual test, in that it initializes the CompletableFuture,
83  	 * attaches the listeners and does the assertions.
84  	 * @throws InterruptedException 
85  	 * @throws ExecutionException 
86  	 * @throws TimeoutException 
87  	 */
88  	public void doTest() throws InterruptedException, ExecutionException, TimeoutException {
89  		// Create a completableFuture to test the background task of the service. This
90  		// completableFuture blocks the test thread until its complete method was called.
91  		CompletableFuture<AnnotationCollection> completableFuture = new CompletableFuture<>();
92  
93  		// Act on the loadService's interesting states.
94  		registerListeners(loadService, completableFuture);
95  
96  		loadService.graphProperty().set(graph);
97  		loadService.start();
98  
99  		// This call blocks the test thread until the completableFuture's complete() method is
100 		// called. Calling get() retrieves the result (e.g., the Graph in this case). Set a
101 		// timeout in case the result does not appear.
102 		AnnotationCollection ac = completableFuture.get(DELAY, TimeUnit.MILLISECONDS);
103 
104 		assertNotNull(ac);
105 		assertEquals(2, ac.getAll().size());
106 	}
