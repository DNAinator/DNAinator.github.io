125 	/**
126 	 * Unit-test the topological ordering.
127 	 */
128 	@Test
129 	public void testTopologicalOrder() {
130 		LinkedList<Integer> order = new LinkedList<>();
131 		try {
132 			EdgeParser ep = new EdgeParserImpl(new BufferedReader(
133 							new InputStreamReader(getEdgeFile(), "UTF-8")));
134 
135 			db.execute(e -> {
136 				for (Node n : new AnalyzeCommand(db.rootIterator()).topologicalOrder(e)) {
137 					order.add(Integer.parseInt((String) n.getProperty(ID.name())));
138 				}
139 			});
140 			while (ep.hasNext()) {
141 				Edge<String> next = ep.next();
142 				int source = Integer.parseInt(next.getSource());
143 				int dest = Integer.parseInt(next.getDest());
144 				assertThat(order.indexOf(source), lessThan(order.indexOf(dest)));
145 			}
146 		} catch (NumberFormatException e) {
147 			fail("Number format incorrect " + e.getMessage());
148 			e.printStackTrace();
149 		} catch (InvalidEdgeFormatException e) {
150 			fail("Edge format incorrect " + e.getMessage());
151 			e.printStackTrace();
152 		} catch (IOException e) {
153 			fail("Error during I/O " + e.getMessage());
154 			e.printStackTrace();
155 		}
156 	}
