17  /**
18   * Test generation of a graph representing a binary tree.
19   */
20  @RunWith(MockitoJUnitRunner.class)
21  public class TreeGeneratorTest {
22  	@Mock private Sink sink;
23  	private TreeGenerator g;
24  	
25  	/**
26  	 * Generic initialization.
27  	 */
28  	@Before
29  	public void setup() {
30  		g = new TreeGenerator();
31  		g.addSink(sink);
32  	}
33  	
34  	/**
35  	 * Test the begin call of a generator.
36  	 */
37  	@Test
38  	public void testBegin() {
39  		g.begin();
40  		Mockito.verify(sink, atLeastOnce()).nodeAdded(any(), anyLong(), any());
41  	}
42  	
43  	/**
44  	 * Test 2 consecutive next calls of a generator.
45  	 */
46  	@Test
47  	public void testNext() {
48  		// CHECKSTYLE.OFF: MagicNumber
49  		g.begin();
50  		g.nextEvents();
51  		Mockito.verify(sink, times(2)).nodeAdded(any(), anyLong(), any());
52  		Mockito.verify(sink, times(1)).edgeAdded(any(), anyLong(), any(),
53  												 any(), any(), anyBoolean());
54  		
55  		g.nextEvents();
56  		Mockito.verify(sink, times(3)).nodeAdded(any(), anyLong(), any());
57  		Mockito.verify(sink, times(2)).edgeAdded(any(), anyLong(), any(),
58  												 any(), any(), anyBoolean());
59  		// CHECKSTYLE.ON: MagicNumber
60  	}
