73  	/**
74  	 * Unit-test the topological ordering.
75  	 */
76  	@Test
77  	public void testTopologicalOrder() {
78  		LinkedList<Integer> order = new LinkedList<>();
79  		try {
80  			EdgeParser ep = new EdgeParserImpl(new BufferedReader(new FileReader(edgeFile)));
81  
82  			try (Transaction tx = service.beginTx()) {
83  				for (Node n : db.topologicalOrder()) {
84  					order.add(Integer.parseInt((String) n.getProperty("id")));
85  				}
86  			}
87  			while (ep.hasNext()) {
88  				Edge<String> next = ep.next();
89  				int source = Integer.parseInt(next.getSource());
90  				int dest = Integer.parseInt(next.getDest());
91  				assertThat(order.indexOf(source), lessThan(order.indexOf(dest)));
92  			}
93  		} catch (NumberFormatException e) {
94  				fail("Number format incorrect " + e.getMessage());
95  				e.printStackTrace();
96  		} catch (InvalidEdgeFormatException e) {
97  				fail("Edge format incorrect " + e.getMessage());
98  				e.printStackTrace();
99  		} catch (IOException e) {
100 				fail("Error during I/O " + e.getMessage());
101 				e.printStackTrace();
102 		}
103 	}
